    int data;

‚¨áÔ∏è Insert at End

üìù Pseudocode

FUNCTION InsertAtEnd(head, value)
    newNode ‚Üê CreateNode(value)
    
    // Empty list case
    IF head = NULL THEN
        RETURN newNode
    END IF
    
    // Find last node
    current ‚Üê head
    WHILE current.next ‚â† NULL
        current ‚Üê current.next
    END WHILE
    
    current.next ‚Üê newNode
    RETURN head
END FUNCTION

‚öôÔ∏è C

void insert_at_end(
    Node **head_ref, 
    int value
) {
    Node *new_node = create_node(value);
    if (new_node == NULL) return;
    
    if (*head_ref == NULL) {
        *head_ref = new_node;
        return;
    }
    
    Node *last = *head_ref;
    while (last->next != NULL) {
        last = last->next;
    }
    last->next = new_node;
}

üêç Python

üóëÔ∏è Delete by Value

üìù Pseudocode

FUNCTION DeleteByValue(head, target)
    IF head = NULL THEN
        RETURN NULL
    END IF
    
    // Head contains target
    IF head.data = target THEN
        temp ‚Üê head.next
        DELETE head
        RETURN temp
    END IF
    
    // Search for target
    current ‚Üê head
    WHILE current.next ‚â† NULL
        IF current.next.data = target
            temp ‚Üê current.next
            current.next ‚Üê temp.next
            DELETE temp
            RETURN head
        END IF
        current ‚Üê current.next
    END WHILE
    RETURN head
END FUNCTION

‚öôÔ∏è C

void delete_by_value(
    Node **head_ref, 
    int target
) {
    if (*head_ref == NULL) return;
    
    // Head case
    if ((*head_ref)->data == target) {
        Node *temp = *head_ref;
        *head_ref = (*head_ref)->next;
        free(temp);  // MUST free!
        return;
    }
    
    Node *curr = *head_ref;
    while (curr->next != NULL) {
        if (curr->next->data == target) {
            Node *temp = curr->next;
            curr->next = temp->next;
            free(temp);
            return;
        }
        curr = curr->next;
    }
}

üêç Python

def delete_by_value(head, target):
    if head is None:
        return None
    
    # Head case
    if head.data == target:
        return head.next  # GC handles memory
    
    current = head
    while current.next:
        if current.next.data == target:
            current.next = current.next.next
            return head
        current = current.next
    
    return head  # Not found

üîÑ Reverse List

üìù Pseudocode

FUNCTION ReverseList(head)
    prev ‚Üê NULL
    current ‚Üê head
    
    WHILE current ‚â† NULL DO
        // Save next
        next ‚Üê current.next
        
        // Reverse link
        current.next ‚Üê prev
        
        // Move forward
        prev ‚Üê current
        current ‚Üê next
    END WHILE
    
    RETURN prev  // New head
END FUNCTION

‚öôÔ∏è C

void reverse_list(Node **head_ref) {
    Node *prev = NULL;
    Node *current = *head_ref;
    Node *next = NULL;
    
    while (current != NULL) {
        next = current->next;
        current->next = prev;
        prev = current;
        current = next;
    }
    
    *head_ref = prev;
}

// Time: O(n), Space: O(1)

üêç Python

def reverse_list(head):
    prev = None
    current = head
    
    while current:
        next_node = current.next
        current.next = prev
        prev = current
        current = next_node
    
    return prev

# Recursive version
def reverse_recursive(head):
    if not head or not head.next:
        return head
    
    new_head = reverse_recursive(head.next)
    head.next.next = head
    head.next = None
    return new_head

üìù Pseudocode

FUNCTION FreeList(head)
    WHILE head ‚â† NULL DO

‚öôÔ∏è C

void free_list(Node **head_ref) {
    Node *current = *head_ref;
    Node *next;
    
    while (current != NULL) {
        next = current->next;
        free(current);  // Manual deallocation
        current = next;
    }
    
    *head_ref = NULL;  // Prevent dangling ptr
}

// CRITICAL: Every malloc needs free!
// Use Valgrind to verify no leaks

üí° Key Insights

1. Algorithm vs Implementation

Pseudocode captures the what; C and Python show the how. Master the algorithm first, then adapt to any language.

2. Memory Matters

C forces you to understand memory layout, allocation and deallocation. This knowledge transfers to all programming.

3. Abstraction Trade-offs

Python's simplicity comes at the cost of control and performance. Choose the right tool for each task.

4. Transferable Skills

Learning linked lists in C gives you deep understanding. The concepts apply everywhere‚Äîjust the syntax changes.
