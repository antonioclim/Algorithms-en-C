üîó Union-Find: Find with Path Compression O(Œ±(n))

Pseudocode Algorithm 
FIND(parent, x):
    // Find root with path compression
    
    if parent[x] ‚â† x then
        // Recursively find root
        parent[x] ‚Üê FIND(parent, parent[x])
    end if
    
    return parent[x]

// After find, all nodes on path
// point directly to root
C Implementation Low-level 
int find(int parent[], int x) {
    if (parent[x] != x) {
        // Path compression:
        // Make all nodes point to root
        parent[x] = find(parent, parent[x]);
    }
    return parent[x];
}

// Usage:
// int parent[N];
// for (int i = 0; i < N; i++)
//     parent[i] = i;  // Init
// int root = find(parent, x);
Python Implementation High-level 
def find(parent, x):
    """Find with path compression."""
    if parent[x] != x:
        # Path compression
        parent[x] = find(parent, parent[x])
    return parent[x]

# Usage:
# parent = list(range(n))  # Init
# root = find(parent, x)

# Alternative (iterative):
def find_iter(parent, x):
    root = x
    while parent[root] != root:
        root = parent[root]
    # Compress path
    while parent[x] != root:
        parent[x], x = root, parent[x]
    return root
Key Insight: Path compression makes all nodes on the find path point directly to the root, flattening the tree structure for future operations.

üîó Union-Find: Union by Rank O(Œ±(n))

Pseudocode Algorithm 
UNION(parent, rank, x, y):
    // Unite sets containing x and y
    
    px ‚Üê FIND(parent, x)
    py ‚Üê FIND(parent, y)
    
    if px = py then
        return  // Already same set
    
    // Attach smaller tree under larger
    if rank[px] < rank[py] then
        parent[px] ‚Üê py
    else if rank[px] > rank[py] then
        parent[py] ‚Üê px
    else
        parent[py] ‚Üê px
        rank[px] ‚Üê rank[px] + 1
    end if
C Implementation Low-level 
void unite(int parent[], int rank[], 
           int x, int y) {
    int px = find(parent, x);
    int py = find(parent, y);
    
    if (px == py) return;
    
    // Union by rank
    if (rank[px] < rank[py]) {
        parent[px] = py;
    } else if (rank[px] > rank[py]) {
        parent[py] = px;
    } else {
        parent[py] = px;
        rank[px]++;
    }
}

// Init: memset(rank, 0, sizeof(rank));
Python Implementation High-level 
def union(parent, rank, x, y):
    """Union by rank."""
    px, py = find(parent, x), find(parent, y)
    
    if px == py:
        return False  # Already united
    
    # Attach smaller tree under larger
    if rank[px] < rank[py]:
        parent[px] = py
    elif rank[px] > rank[py]:
        parent[py] = px
    else:
        parent[py] = px
        rank[px] += 1
    
    return True

# Init: rank = [0] * n

Arrays passed by pointer
Python Feature
Lists mutable in-place
Combined Complexity
O(Œ±(n)) ‚âà O(1)

üå≤ Kruskal's MST Algorithm O(E log E)

Pseudocode Algorithm 
KRUSKAL(V, E):
    // Input: V vertices, E edges
    
    MST ‚Üê empty set
    SORT(E) by weight ascending
    
    // Initialize Union-Find
    for each v in V do
        MAKE-SET(v)
    
    for each edge (u,v,w) in E do
        if FIND(u) ‚â† FIND(v) then
            MST ‚Üê MST ‚à™ {(u,v,w)}
            UNION(u, v)
            if |MST| = |V| - 1 then
                break
    
    return MST
C Implementation Low-level 
typedef struct {
    int u, v, w;
} Edge;

int cmp(const void *a, const void *b) {
    return ((Edge*)a)->w - ((Edge*)b)->w;
}

int kruskal(Edge edges[], int E, int V,
            Edge mst[]) {
    qsort(edges, E, sizeof(Edge), cmp);
    
    int parent[V], rank[V];
    for (int i = 0; i < V; i++) {
        parent[i] = i; rank[i] = 0;
    }
    
    int mst_size = 0, total = 0;
    for (int i = 0; i < E && mst_size < V-1; i++) {
        int u = edges[i].u, v = edges[i].v;
        if (find(parent,u) != find(parent,v)) {
            unite(parent, rank, u, v);
            mst[mst_size++] = edges[i];
            total += edges[i].w;
        }
    }
    return total;
}
Python Implementation High-level 
def kruskal(n, edges):
    """
    edges: list of (u, v, weight)
    Returns: (total_weight, mst_edges)
    """
    # Sort by weight
    edges = sorted(edges, key=lambda x: x[2])
    
    parent = list(range(n))
    rank = [0] * n
    
    mst = []
    total = 0
    
    for u, v, w in edges:
        if find(parent, u) != find(parent, v):
            union(parent, rank, u, v)
            mst.append((u, v, w))
            total += w
            if len(mst) == n - 1:
                break
    
    return total, mst
Note: Python's sorted() uses Timsort (O(n log n)), C's qsort() varies by implementation. Both achieve O(E log E) overall.

üìê Topological Sort (DFS) O(V + E)

Pseudocode Algorithm 
TOPO-SORT-DFS(G):
    // G = (V, E) directed acyclic graph
    
    visited ‚Üê array[|V|] of false
    stack ‚Üê empty stack
    
    for each vertex v in V do
        if not visited[v] then
            DFS-VISIT(v, visited, stack)
    
    return REVERSE(stack)

DFS-VISIT(v, visited, stack):
    visited[v] ‚Üê true
    
    for each neighbour u of v do
        if not visited[u] then
            DFS-VISIT(u, visited, stack)
    
    PUSH(stack, v)  // Post-order
C Implementation Low-level 
void dfs_visit(int v, int adj[][MAX], int V,
               bool visited[], int stack[], 
               int *top) {
    visited[v] = true;
    
    for (int u = 0; u < V; u++) {
        if (adj[v][u] && !visited[u]) {
            dfs_visit(u, adj, V, visited, 
                      stack, top);
        }
    }
    stack[(*top)++] = v;  // Post-order
}

void topo_sort(int adj[][MAX], int V, 
               int result[]) {
    bool visited[MAX] = {false};
    int stack[MAX], top = 0;
    
    for (int v = 0; v < V; v++) {
        if (!visited[v])
            dfs_visit(v, adj, V, visited, 
                      stack, &top);
    }
    // Reverse stack into result
    for (int i = 0; i < V; i++)
        result[i] = stack[V - 1 - i];
}
Python Implementation High-level 
def topo_sort_dfs(graph):
    """
    graph: dict {v: [neighbours]}
    Returns: topological order list
    """
    visited = set()
    stack = []
    
    def dfs(v):
        visited.add(v)
        for u in graph.get(v, []):
            if u not in visited:
                dfs(u)
        stack.append(v)  # Post-order
    
    for v in graph:
        if v not in visited:
            dfs(v)
    
    return stack[::-1]  # Reverse

# Usage:
# graph = {0: [1, 2], 1: [3], 2: [3], 3: []}
# order = topo_sort_dfs(graph)

Python Style

üìä Kahn's Algorithm (BFS Topological Sort) O(V + E)

Pseudocode Algorithm 
KAHN(G):
    // G = (V, E) directed graph
    
    in_degree ‚Üê array[|V|] of 0
    
    // Calculate in-degrees
    for each edge (u,v) in E do
        in_degree[v] ‚Üê in_degree[v] + 1
    
    queue ‚Üê vertices with in_degree = 0
    result ‚Üê empty list
    
    while queue not empty do
        u ‚Üê DEQUEUE(queue)
        APPEND(result, u)
        
        for each neighbour v of u do
            in_degree[v] ‚Üê in_degree[v] - 1
            if in_degree[v] = 0 then
                ENQUEUE(queue, v)
    
    if |result| ‚â† |V| then
        return "CYCLE DETECTED"
    return result
C Implementation Low-level 
bool kahn(int adj[][MAX], int V, int result[]) {
    int in_degree[MAX] = {0};
    int queue[MAX], front = 0, rear = 0;
    
    // Calculate in-degrees
    for (int u = 0; u < V; u++)
        for (int v = 0; v < V; v++)
            if (adj[u][v]) in_degree[v]++;
    
    // Enqueue vertices with 0 in-degree
    for (int v = 0; v < V; v++)
        if (in_degree[v] == 0)
            queue[rear++] = v;
    
    int count = 0;
    while (front < rear) {
        int u = queue[front++];
        result[count++] = u;
        
        for (int v = 0; v < V; v++) {
            if (adj[u][v]) {
                in_degree[v]--;
                if (in_degree[v] == 0)
                    queue[rear++] = v;
            }
        }
    }
    return count == V;  // false = cycle
}
Python Implementation High-level 
from collections import deque

def kahn(graph, n):
    """
    graph: dict {v: [neighbours]}
    n: number of vertices
    Returns: (success, order) or (False, [])
    """
    in_degree = [0] * n
    
    for u in graph:
        for v in graph[u]:
            in_degree[v] += 1
    
    queue = deque([v for v in range(n) 
                   if in_degree[v] == 0])
    result = []
    
    while queue:
        u = queue.popleft()
        result.append(u)
        
        for v in graph.get(u, []):
            in_degree[v] -= 1
            if in_degree[v] == 0:
                queue.append(v)
    
    if len(result) != n:
        return False, []  # Cycle!
    return True, result
Cycle Detection Bonus: If result length ‚â† V, the graph contains a cycle and no topological ordering exists!

‚ôõ N-Queens: Safety Check O(n)

Pseudocode Algorithm 
IS-SAFE(board, row, col):
    // board[i] = column of queen in row i
    // Check if placing queen at (row, col) is safe
    
    for i ‚Üê 0 to row - 1 do
        // Same column?
        if board[i] = col then
            return false
        
        // Same diagonal?
        // |row_diff| = |col_diff| means diagonal
        if |board[i] - col| = |i - row| then
            return false
    
    return true

// Diagonal insight:
// ‚Üò diagonal: row - col is constant
// ‚Üô diagonal: row + col is constant
C Implementation Low-level 
#include <stdlib.h>  // for abs()

bool is_safe(int board[], int row, int col) {
    for (int i = 0; i < row; i++) {
        // Same column check
        if (board[i] == col)
            return false;
        
        // Diagonal check:
        // If |row_diff| == |col_diff|,
        // queens are on same diagonal
        if (abs(board[i] - col) == abs(i - row))
            return false;
    }
    return true;
}

‚ôõ N-Queens: Backtracking Solver O(n!)

Pseudocode Algorithm 
SOLVE-NQUEENS(board, row, n):
    // Place queens row by row
    
    if row = n then
        // All queens placed!
        PRINT-SOLUTION(board, n)
        count ‚Üê count + 1
        return
    
    for col ‚Üê 0 to n - 1 do
        if IS-SAFE(board, row, col) then
            board[row] ‚Üê col      // Place
            SOLVE-NQUEENS(board, row + 1, n)
            // Backtrack is implicit:
            // next iteration overwrites board[row]

NQUEENS(n):
    board ‚Üê array[n]
    count ‚Üê 0
    SOLVE-NQUEENS(board, 0, n)
    return count

def solve_nqueens(n):
    """Find all N-Queens solutions."""
    solutions = []
    
    def solve(board, row):
        if row == n:
            solutions.append(board[:])
            return
        
        for col in range(n):
            if is_safe(board, row, col):
                board[row] = col
                solve(board, row + 1)
                # Implicit backtrack
    
    solve([0] * n, 0)
    return solutions

Pseudocode Algorithm 
IS-VALID(grid, row, col, num):
    // Check if placing num at (row,col) is valid
    
    // Check row
    for c ‚Üê 0 to 8 do
        if grid[row][c] = num then

C Implementation Low-level 
bool is_valid(int grid[9][9], int row, 
              int col, int num) {
    // Check row
    for (int c = 0; c < 9; c++)
        if (grid[row][c] == num)
            return false;
    
    // Check column
    for (int r = 0; r < 9; r++)
        if (grid[r][col] == num)
            return false;
    
    // Check 3x3 box
    int box_r = (row / 3) * 3;
    int box_c = (col / 3) * 3;
    for (int r = 0; r < 3; r++)
        for (int c = 0; c < 3; c++)
            if (grid[box_r+r][box_c+c] == num)
                return false;
    
    return true;
}
Python Implementation High-level 
def is_valid(grid, row, col, num):
    """Check if num can be placed at (row, col)."""
    # Check row
    if num in grid[row]:
        return False
    
    # Check column
    if num in [grid[r][col] for r in range(9)]:
        return False
    
    # Check 3x3 box
    box_r, box_c = 3 * (row // 3), 3 * (col // 3)
    for r in range(box_r, box_r + 3):
        for c in range(box_c, box_c + 3):
            if grid[r][c] == num:
                return False
    
    return True

üî¢ Sudoku: Backtracking Solver O(9^n)

Pseudocode Algorithm 
SOLVE-SUDOKU(grid):
    // Find empty cell
    (row, col) ‚Üê FIND-EMPTY(grid)
    if no empty cell then
        return true  // Solved!
    
    for num ‚Üê 1 to 9 do
        if IS-VALID(grid, row, col, num) then
            grid[row][col] ‚Üê num
            
            if SOLVE-SUDOKU(grid) then
                return true
            
            grid[row][col] ‚Üê 0  // Backtrack
    
    return false  // Trigger backtrack

FIND-EMPTY(grid):
    for r ‚Üê 0 to 8 do
        for c ‚Üê 0 to 8 do
            if grid[r][c] = 0 then
                return (r, c)
    return null
C Implementation Low-level 
bool find_empty(int grid[9][9], 
                int *row, int *col) {
    for (int r = 0; r < 9; r++) {
        for (int c = 0; c < 9; c++) {
            if (grid[r][c] == 0) {
                *row = r; *col = c;
                return true;
            }
        }
    }
    return false;
}

bool solve_sudoku(int grid[9][9]) {
    int row, col;
    if (!find_empty(grid, &row, &col))
        return true;  // Solved!
    
    for (int num = 1; num <= 9; num++) {
        if (is_valid(grid, row, col, num)) {
            grid[row][col] = num;
            if (solve_sudoku(grid))
                return true;
            grid[row][col] = 0;  // Backtrack
        }
    }
    return false;
}
Python Implementation High-level 
def solve_sudoku(grid):
    """Solve Sudoku using backtracking."""
    # Find empty cell
    empty = find_empty(grid)
    if not empty:
        return True  # Solved!
    
    row, col = empty
    
    for num in range(1, 10):
        if is_valid(grid, row, col, num):
            grid[row][col] = num
            
            if solve_sudoku(grid):
                return True
            
            grid[row][col] = 0  # Backtrack
    
    return False

def find_empty(grid):
    for r in range(9):
        for c in range(9):
            if grid[r][c] == 0:
                return (r, c)
    return None
Performance: Easy puzzles: ~100 backtracks. Hard "evil" puzzles: ~10,000+ backtracks. MRV heuristic dramatically improves!
