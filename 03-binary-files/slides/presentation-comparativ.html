Opening a Binary File

üìù Pseudocode

FUNCTION open_file(path, mode):
    file ‚Üê OPEN path IN mode
    IF file IS NULL THEN
        RETURN error
    END IF
    RETURN file
END FUNCTION

// Usage
file ‚Üê open_file("data.bin", WRITE_BINARY)

‚öôÔ∏è C

FILE* open_file(const char* path, 
                const char* mode) {
    FILE *fp = fopen(path, mode);
    if (fp == NULL) {
        perror("Error");
        return NULL;
    }
    return fp;
}

// Usage
FILE *fp = open_file("data.bin", "wb");

üêç Python

def open_file(path, mode):
    try:
        return open(path, mode)
    except IOError as e:
        print(f"Error: {e}")
        return None

# Usage (preferred: context manager)
with open("data.bin", "wb") as f:
    # file auto-closes
    pass
Key Difference: Python's with statement ensures automatic resource cleanup, eliminating a common source of bugs in C.

Writing a Record to Binary File

üìù Pseudocode

FUNCTION write_record(file, record):
    bytes ‚Üê SERIALISE(record)
    count ‚Üê WRITE bytes TO file
    
    IF count ‚â† SIZE_OF(record) THEN
        RETURN error
    END IF
    
    RETURN success
END FUNCTION

‚öôÔ∏è C

int write_record(FILE *fp, 
                 const Student *s) {
    size_t written = fwrite(
        s,              // data pointer
        sizeof(Student),// element size
        1,              // count
        fp              // file
    );
    return (written == 1) ? 0 : -1;
}

// Usage
write_record(fp, &student);

üêç Python

def write_record(f, student):
    # Encode string to bytes
    name_bytes = student.name.encode(
        'utf-8')[:50].ljust(50, b'\x00')
    
    # Pack into binary format
    data = struct.pack(
        "i50sfi",
        student.id,
        name_bytes,
        student.gpa,
        student.year
    )
    f.write(data)

Reading Records from Binary File

üìù Pseudocode

FUNCTION read_all_records(file):
    records ‚Üê EMPTY LIST
    
    WHILE NOT end_of_file(file):
        bytes ‚Üê READ SIZE_OF(Record) 
                FROM file
        IF bytes IS EMPTY THEN
            BREAK
        END IF
        record ‚Üê DESERIALISE(bytes)
        APPEND record TO records
    END WHILE
    
    RETURN records
END FUNCTION

‚öôÔ∏è C

int read_all(FILE *fp, Student **out,
             size_t *count) {
    // Get file size
    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    fseek(fp, 0, SEEK_SET);
    
    *count = size / sizeof(Student);
    *out = malloc(*count * sizeof(Student));
    
    size_t read = fread(*out, 
        sizeof(Student), *count, fp);
    
    return (read == *count) ? 0 : -1;
}

üêç Python

def read_all_records(f):
    records = []
    record_size = struct.calcsize("i50sfi")
    
    while True:
        data = f.read(record_size)
        if len(data) < record_size:
            break
        
        id, name, gpa, year = struct.unpack(
            "i50sfi", data)
        
        name = name.rstrip(b'\x00').decode()
        records.append(
            Student(id, name, gpa, year))
    
    return records

Random Access: Seeking to Record N

üìù Pseudocode

FUNCTION get_record(file, index):
    offset ‚Üê index √ó SIZE_OF(Record)
    
    SEEK file TO offset FROM START
    
    bytes ‚Üê READ SIZE_OF(Record) 
            FROM file
    
    IF bytes IS EMPTY THEN
        RETURN NULL
    END IF
    
    RETURN DESERIALISE(bytes)
END FUNCTION

‚öôÔ∏è C

Student* get_record(FILE *fp, 
                    size_t index) {
    static Student s;
    
    long offset = index * sizeof(Student);
    
    if (fseek(fp, offset, SEEK_SET) != 0)
        return NULL;
    
    if (fread(&s, sizeof(Student), 1, fp) != 1)
        return NULL;
    
    return &s;
}

// Access record at index 5
Student *s = get_record(fp, 5);

üêç Python

def get_record(f, index):
    record_size = struct.calcsize("i50sfi")
    offset = index * record_size
    
    f.seek(offset)  # SEEK_SET is default
    
    data = f.read(record_size)
    if len(data) < record_size:
        return None
    
    id, name, gpa, year = struct.unpack(
        "i50sfi", data)
    
    name = name.rstrip(b'\x00').decode()
    return Student(id, name, gpa, year)

# Access record at index 5
s = get_record(f, 5)
O(1) Access: All three approaches calculate the exact byte offset, enabling constant-time access regardless of file size.

Updating a Record In-Place

üìù Pseudocode

FUNCTION update_record(file, index, 
                       new_data):
    offset ‚Üê index √ó SIZE_OF(Record)
    
    // Seek to position
    SEEK file TO offset FROM START
    
    // Read existing (optional)
    old ‚Üê READ record FROM file
    
    // Return to same position
    SEEK file TO offset FROM START
    
    // Write new data
    WRITE new_data TO file
    
    FLUSH file
END FUNCTION

‚öôÔ∏è C

int update_record(FILE *fp, size_t index,
                  const Student *new_s) {
    long offset = index * sizeof(Student);
    
    // Open in read+write mode: "r+b"
    if (fseek(fp, offset, SEEK_SET) != 0)
        return -1;
    
    if (fwrite(new_s, sizeof(Student), 
               1, fp) != 1)
        return -1;
    
    fflush(fp);  // Ensure write completes
    return 0;
}

üêç Python

def update_record(f, index, new_student):
    record_size = struct.calcsize("i50sfi")
    offset = index * record_size
    
    f.seek(offset)
    
    name_bytes = new_student.name.encode(
        'utf-8')[:50].ljust(50, b'\x00')
    
    data = struct.pack(
        "i50sfi",
        new_student.id,
        name_bytes,
        new_student.gpa,
        new_student.year
    )
    
    f.write(data)
    f.flush()  # Force write to disk

Calculating File Size and Record Count

üìù Pseudocode

FUNCTION count_records(file):
    // Save current position
    current ‚Üê GET_POSITION(file)
    
    // Seek to end
    SEEK file TO 0 FROM END
    file_size ‚Üê GET_POSITION(file)
    
    // Restore position
    SEEK file TO current FROM START
    
    RETURN file_size √∑ SIZE_OF(Record)
END FUNCTION

‚öôÔ∏è C

size_t count_records(FILE *fp) {
    // Save position
    long current = ftell(fp);
    
    // Get file size
    fseek(fp, 0, SEEK_END);
    long file_size = ftell(fp);
    
    // Restore position
    fseek(fp, current, SEEK_SET);
    
    return file_size / sizeof(Student);
}

// Alternative using stat()
#include <sys/stat.h>
struct stat st;
stat("data.bin", &st);
size_t count = st.st_size / sizeof(Student);

üêç Python

def count_records(f):
    # Save position
    current = f.tell()
    
    # Get file size
    f.seek(0, 2)  # 2 = SEEK_END
    file_size = f.tell()
    
    # Restore position
    f.seek(current)
    
    record_size = struct.calcsize("i50sfi")
    return file_size // record_size

# Alternative using os.path
import os
size = os.path.getsize("data.bin")
count = size // struct.calcsize("i50sfi")

Error Handling Patterns

üìù Pseudocode

FUNCTION safe_read(file):
    TRY:
        data ‚Üê READ FROM file
        IF read_failed THEN
            IF end_of_file(file) THEN
                RETURN EOF_MARKER
            ELSE
                THROW ReadError
            END IF
        END IF
        RETURN data
    CATCH error:
        LOG error
        RETURN NULL
    END TRY
END FUNCTION

‚öôÔ∏è C

int safe_read(FILE *fp, Student *s) {
    size_t read = fread(s, sizeof(*s), 
                        1, fp);
    
    if (read != 1) {
        if (feof(fp)) {
            return 1;  // EOF, not error
        }
        if (ferror(fp)) {
            perror("Read error");
            clearerr(fp);  // Reset
            return -1;  // Error
        }
    }
    return 0;  // Success
}

üêç Python

def safe_read(f, record_size):
    try:
        data = f.read(record_size)
        
        if len(data) == 0:
            return None  # EOF
        
        if len(data) < record_size:
            raise IOError(
                "Incomplete record")
        
        return data
        
    except IOError as e:
        print(f"Read error: {e}")
        return None
Pattern: Always distinguish between EOF (normal termination) and actual errors (disk failure, permission issues).

Complete Mini-Database Example

Putting it all together

‚öôÔ∏è C Implementation

// Simple student database
int main(void) {
    FILE *fp = fopen("students.bin", "w+b");
    
    // Write some records
    Student students[] = {
        {1001, "Alice", 3.75f, 2024},
        {1002, "Bob", 3.50f, 2023},
        {1003, "Carol", 3.90f, 2024}
    };
    
    fwrite(students, sizeof(Student), 3, fp);
    
    // Random access: read record 1
    fseek(fp, sizeof(Student), SEEK_SET);
    Student s;
    fread(&s, sizeof(Student), 1, fp);
    printf("Record 1: %s\n", s.name);
    
    // Update record 0
    students[0].gpa = 3.80f;
    fseek(fp, 0, SEEK_SET);
    fwrite(&students[0], sizeof(Student), 1, fp);
    
    fclose(fp);
    return 0;
}

üêç Python Implementation

import struct

FMT = "i50sfi"
SIZE = struct.calcsize(FMT)

def pack_student(s):
    name = s.name.encode()[:50].ljust(50, b'\x00')
    return struct.pack(FMT, s.id, name, s.gpa, s.year)

def unpack_student(data):
    id, name, gpa, year = struct.unpack(FMT, data)
    return Student(id, name.rstrip(b'\x00').decode(), 
                   gpa, year)

# Usage
with open("students.bin", "w+b") as f:
    # Write records
    for s in [Student(1001, "Alice", 3.75, 2024),
              Student(1002, "Bob", 3.50, 2023)]:
        f.write(pack_student(s))
    
    # Random access
    f.seek(SIZE)
    data = f.read(SIZE)
    print(unpack_student(data))

üí° Key Insights

1. Concepts Transfer

Binary file operations follow the same logic across languages: open, seek, read/write, close. Master the concept, and syntax becomes secondary.

2. Abstraction Has Costs

Python's convenience (automatic memory, exceptions, with statement) comes with performance overhead. C's verbosity gives maximum control.

3. Binary Formats Require Care

Cross-platform binary files need explicit handling of endianness, padding and type sizes. Python's struct module mirrors C's memory layout.

4. Pseudocode First

Design your algorithm in pseudocode before implementing. This clarifies logic and makes translation to any language straightforward.

Summary

Binary Files Across Languages

Core Pattern:

  1. Define fixed-size record structure
  2. Open file in binary mode
  3. Serialise/deserialise records
  4. Use seek for random access
  5. Handle errors appropriately
  6. Close file (or use context manager)
