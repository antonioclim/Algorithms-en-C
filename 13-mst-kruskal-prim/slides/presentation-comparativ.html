‚ö° The Relaxation Operation

Core operation for shortest paths

üìù

Pseudocode

RELAX(u, v, w):
  // Can we improve path to v via u?
  if dist[u] ‚â† ‚àû and 
     dist[u] + w < dist[v]:
    
    dist[v] ‚Üê dist[u] + w
    parent[v] ‚Üê u
    return TRUE
  
  return FALSE
‚öôÔ∏è

C

#include <limits.h>

int relax(int u, int v, int w,
          int dist[], int parent[]) {
    // Check for overflow
    if (dist[u] == INT_MAX)
        return 0;
    
    if (dist[u] + w < dist[v]) {
        dist[v] = dist[u] + w;
        parent[v] = u;
        return 1;  // Improved
    }
    return 0;  // No change
}

import math

    """Try to improve path to v via u"""
    if dist[u] + w < dist[v]:
        dist[v] = dist[u] + w
        parent[v] = u

Key insight: C requires explicit overflow checking with INT_MAX. Python's float('inf') handles infinity arithmetic naturally.

üî∑ Dijkstra's Algorithm

Complete implementation comparison

üìù

Pseudocode

DIJKSTRA(G, src):
  for v ‚àà G.V:
    dist[v] ‚Üê ‚àû
    parent[v] ‚Üê NIL
  dist[src] ‚Üê 0
  
  Q ‚Üê PRIORITY-QUEUE(G.V)
  
  while Q not empty:
    u ‚Üê EXTRACT-MIN(Q)
    
    for each (v, w) in G.adj[u]:
      if RELAX(u, v, w):
        DECREASE-KEY(Q, v, dist[v])
  
  return dist, parent
‚öôÔ∏è

C

void dijkstra(Graph *g, int src,
              int dist[], int parent[]) {
    int V = g->V;
    MinHeap *pq = createHeap(V);
    
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        parent[i] = -1;
        insertHeap(pq, i, INT_MAX);
    }
    dist[src] = 0;
    decreaseKey(pq, src, 0);
    
    while (pq->size > 0) {
        HeapNode min = extractMin(pq);
        int u = min.vertex;
        
        for (Edge *e = g->adj[u]; 
             e != NULL; e = e->next) {
            int v = e->dest;
            int w = e->weight;
            if (relax(u, v, w, dist, parent))
                decreaseKey(pq, v, dist[v]);
        }
    }
    freeHeap(pq);
}
üêç

Python

import heapq

def dijkstra(graph, src):
    V = len(graph)
    dist = {v: float('inf') for v in graph}
    parent = {v: None for v in graph}
    dist[src] = 0
    
    # (distance, vertex) tuples
    pq = [(0, src)]
    
    while pq:
        d, u = heapq.heappop(pq)
        
        # Skip if outdated entry
        if d > dist[u]:
            continue
            
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                parent[v] = u
                heapq.heappush(pq, 
                    (dist[v], v))
    
    return dist, parent

üî∫ Priority Queue Implementation

Min-heap with decrease-key

üìù

Pseudocode

MIN-HEAP:
  nodes: array of (vertex, dist)
  position: array mapping v ‚Üí index
  size: integer

EXTRACT-MIN(H):
  min ‚Üê H.nodes[0]
  H.nodes[0] ‚Üê H.nodes[H.size - 1]
  H.size ‚Üê H.size - 1
  HEAPIFY-DOWN(H, 0)
  return min

DECREASE-KEY(H, v, newDist):
  idx ‚Üê H.position[v]
  H.nodes[idx].dist ‚Üê newDist
  HEAPIFY-UP(H, idx)
‚öôÔ∏è

C

typedef struct {
    int vertex, distance;
} HeapNode;

typedef struct {
    HeapNode *nodes;
    int *position;
    int size, capacity;
} MinHeap;

void decreaseKey(MinHeap *h, 
                 int v, int d) {
    int idx = h->position[v];
    h->nodes[idx].distance = d;
    
    // Bubble up
    while (idx > 0) {
        int p = (idx - 1) / 2;
        if (h->nodes[idx].distance 
            < h->nodes[p].distance) {
            swap(h, idx, p);
            idx = p;
        } else break;
    }
}

üî∂ Bellman-Ford Algorithm

Handles negative weights

üìù

Pseudocode

BELLMAN-FORD(G, src):
  for v ‚àà G.V:
    dist[v] ‚Üê ‚àû
    parent[v] ‚Üê NIL
  dist[src] ‚Üê 0
  
  // Relax all edges V-1 times
  for i ‚Üê 1 to |V| - 1:
    for each edge (u, v, w) in G.E:
      RELAX(u, v, w)
  
  // Check for negative cycle
  for each edge (u, v, w) in G.E:
    if dist[u] + w < dist[v]:
      return "NEGATIVE CYCLE"
  
  return dist, parent
‚öôÔ∏è

C

int bellmanFord(int V, Edge edges[],
                int E, int src,
                int dist[], int parent[]) {
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        parent[i] = -1;
    }
    dist[src] = 0;
    
    // V-1 iterations
    for (int i = 0; i < V - 1; i++) {
        int changed = 0;
        for (int j = 0; j < E; j++) {
            if (relax(edges[j].src,
                      edges[j].dest,
                      edges[j].weight,
                      dist, parent))
                changed = 1;
        }
        if (!changed) break; // Early exit
    }
    
    // Negative cycle check
    for (int j = 0; j < E; j++)
        if (dist[edges[j].src] != INT_MAX &&
            dist[edges[j].src] + 
            edges[j].weight < 
            dist[edges[j].dest])
            return 0;  // Cycle!
    return 1;
}
üêç

Python

def bellman_ford(graph, edges, src):
    V = len(graph)
    dist = {v: float('inf') for v in graph}
    parent = {v: None for v in graph}
    dist[src] = 0
    
    # V-1 iterations
    for _ in range(V - 1):
        changed = False
        for u, v, w in edges:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                parent[v] = u
                changed = True
        if not changed:
            break  # Early termination
    
    # Check for negative cycle
    for u, v, w in edges:
        if dist[u] + w < dist[v]:
            raise ValueError(
                "Negative cycle detected")
    
    return dist, parent

// Iterative with stack
void getPath(int parent[], int dest,
             int path[], int *len) {
    *len = 0;
    int v = dest;
    while (v != -1) {
        path[(*len)++] = v;
        v = parent[v];
    }
    // Reverse the path
    for (int i = 0; i < *len/2; i++) {
        int t = path[i];
        path[i] = path[*len-1-i];
        path[*len-1-i] = t;
    }
}

def get_path(parent, dest):
    """Reconstruct path from parent dict"""
    path = []
    v = dest
    while v is not None:
        path.append(v)
        v = parent[v]
    return path[::-1]  # Reverse

def print_path(parent, dest):
    """Print path with arrows"""
    path = get_path(parent, dest)
    print(" ‚Üí ".join(map(str, path)))

C

// Manual memory management
Graph* createGraph(int V) {
    Graph *g = malloc(sizeof(Graph));
    g->V = V;
    g->adj = calloc(V, sizeof(Edge*));
    return g;
}

void freeGraph(Graph *g) {
    for (int i = 0; i < g->V; i++) {
        Edge *e = g->adj[i];
        while (e) {
            Edge *next = e->next;
            free(e);
            e = next;
        }
    }
    free(g->adj);
    free(g);
}

// MUST call freeGraph() or leak!

Key Insights

1. The algorithm is the same ‚Äî syntax differs, logic remains.

3. Python trades performance for convenience ‚Äî heapq lacks decrease-key.

4. Pseudocode abstracts implementation details ‚Äî focus on correctness first.

5. Understanding all three makes you a better problem solver.
