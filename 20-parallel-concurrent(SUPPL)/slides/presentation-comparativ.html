Condition Variables â€” Signalling Between Threads

ğŸ“ Pseudocode
SHARED data_ready â† FALSE
MUTEX lock
CONDITION cond

FUNCTION producer():
    // Produce data
    data â† generate_data()
    
    ACQUIRE(lock)
    shared_data â† data
    data_ready â† TRUE
    SIGNAL(cond)
    RELEASE(lock)

    ACQUIRE(lock)
    
    WHILE NOT data_ready:
        WAIT(cond, lock)  // Releases lock
    
    // Consume data
    result â† process(shared_data)
    RELEASE(lock)

WHILE not IF! Spurious wakeups can occur.

Producer-Consumer with Bounded Buffer

ğŸ“ Pseudocode
CONSTANT CAPACITY â† 10
SHARED buffer[CAPACITY]
SHARED count â† 0
MUTEX lock
CONDITION not_full
CONDITION not_empty

FUNCTION produce(item):
    ACQUIRE(lock)
    WHILE count = CAPACITY:
        WAIT(not_full, lock)
    
    buffer[count] â† item
    count â† count + 1
    SIGNAL(not_empty)
    RELEASE(lock)

FUNCTION consume():
    ACQUIRE(lock)
    WHILE count = 0:
        WAIT(not_empty, lock)
    
    count â† count - 1
    item â† buffer[count]
    SIGNAL(not_full)
    RELEASE(lock)
    RETURN item
âš™ï¸ C (pthreads)
#define CAPACITY 10

typedef struct {
    int buffer[CAPACITY];
    int count;
    pthread_mutex_t lock;
    pthread_cond_t not_full;
    pthread_cond_t not_empty;
} BoundedBuffer;

void produce(BoundedBuffer *bb, 
             int item) {
    pthread_mutex_lock(&bb->lock);
    while (bb->count == CAPACITY)
        pthread_cond_wait(
            &bb->not_full, &bb->lock);
    
    bb->buffer[bb->count++] = item;
    pthread_cond_signal(&bb->not_empty);
    pthread_mutex_unlock(&bb->lock);
}

int consume(BoundedBuffer *bb) {
    pthread_mutex_lock(&bb->lock);
    while (bb->count == 0)
        pthread_cond_wait(
            &bb->not_empty, &bb->lock);
    
    int item = bb->buffer[--bb->count];
    pthread_cond_signal(&bb->not_full);
    pthread_mutex_unlock(&bb->lock);
    return item;
}

Semaphore vs Mutex: Semaphores can be signalled by any thread, mutexes should be unlocked by the same thread that locked them.

âš™ï¸ C (POSIX)

Thread Pool Pattern

ğŸ“ Pseudocode
CLASS ThreadPool:
    workers[N]
    task_queue
    running â† TRUE
    
    FUNCTION worker_loop():
        WHILE running:
            task â† queue.dequeue()  // Blocks
            IF task â‰  NULL:
                task.execute()
    
    FUNCTION submit(task):
        queue.enqueue(task)
    
    FUNCTION shutdown():
        running â† FALSE
        // Signal all workers
        FOR EACH worker IN workers:
            queue.enqueue(NULL)  // Poison pill
        FOR EACH worker IN workers:
            JOIN(worker)
âš™ï¸ C (Manual)
typedef struct {
    void (*func)(void *);
    void *arg;
} Task;

typedef struct {
    pthread_t *threads;
    size_t num_threads;
    Task *queue;
    size_t queue_size, head, tail;
    pthread_mutex_t lock;
    pthread_cond_t not_empty;
    int shutdown;
} ThreadPool;

void *worker(void *arg) {
    ThreadPool *pool = arg;
    
    while (1) {
        pthread_mutex_lock(&pool->lock);
        while (pool->head == pool->tail 
               && !pool->shutdown) {
            pthread_cond_wait(
                &pool->not_empty, 
                &pool->lock);
        }
        if (pool->shutdown) {
            pthread_mutex_unlock(&pool->lock);
            break;
        }
        Task task = pool->queue[pool->head++];
        pthread_mutex_unlock(&pool->lock);
        task.func(task.arg);
    }
    return NULL;
}

Parallel Merge Sort

ğŸ“ Pseudocode
FUNCTION parallel_merge_sort(arr, depth):
    IF length(arr) â‰¤ 1:
        RETURN arr
    
    mid â† length(arr) / 2
    left â† arr[0..mid]
    right â† arr[mid..end]
    
    IF depth < MAX_DEPTH:
        // Parallel recursion
        t1 â† SPAWN sort(left, depth+1)
        t2 â† SPAWN sort(right, depth+1)
        left â† SYNC(t1)
        right â† SYNC(t2)
    ELSE:
        // Sequential below threshold
        left â† sort(left, depth+1)
        right â† sort(right, depth+1)
    
    RETURN merge(left, right)

Depth limit: Prevents thread explosion (2^depth threads at depth d)

âš™ï¸ C (pthreads)
#define MAX_DEPTH 3

typedef struct {
    int *arr, *temp;
    int left, right, depth;
} SortArgs;

void *parallel_sort(void *arg) {
    SortArgs *a = arg;
    if (a->right - a->left <= 1) return NULL;
    
    int mid = (a->left + a->right) / 2;
    
    if (a->depth < MAX_DEPTH) {
        SortArgs larg = {a->arr, a->temp, 
            a->left, mid, a->depth + 1};
        SortArgs rarg = {a->arr, a->temp, 
            mid, a->right, a->depth + 1};
        
        pthread_t t1, t2;
        pthread_create(&t1, NULL, 
            parallel_sort, &larg);
        pthread_create(&t2, NULL, 
            parallel_sort, &rarg);
        pthread_join(t1, NULL);
        pthread_join(t2, NULL);
    } else {
        // Sequential
        merge_sort_seq(a->arr, a->temp,
            a->left, mid);
        merge_sort_seq(a->arr, a->temp,
            mid, a->right);
    }
    merge(a->arr, a->temp, 
          a->left, mid, a->right);
    return NULL;
}

from multiprocessing import Pool
import numpy as np

def merge(left, right):
    result = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    result.extend(left[i:])
    result.extend(right[j:])
    return result

def parallel_sort(arr, pool, depth=0):
    if len(arr) <= 1:
        return arr
    
    mid = len(arr) // 2
    
    if depth < 2:  # Limit parallelism
        left = pool.apply_async(
            parallel_sort, 
            (arr[:mid], pool, depth+1))
        right = pool.apply_async(
            parallel_sort,
            (arr[mid:], pool, depth+1))
        return merge(left.get(), right.get())
    else:
        return merge(
            sorted(arr[:mid]),
            sorted(arr[mid:])
        )
