Hash Function (djb2 Algorithm)

üìù Pseudocode
FUNCTION hash(key):
    hash_value ‚Üê 5381
    
    FOR EACH character c IN key:
        hash_value ‚Üê hash_value √ó 33 + c
    END FOR
    
    RETURN hash_value
END FUNCTION

// Usage:
index ‚Üê hash(key) MOD table_size
‚öôÔ∏è C
unsigned int hash_djb2(const char *key) {
    unsigned int hash = 5381;
    int c;
    
    while ((c = *key++)) {
        // hash * 33 + c
        hash = ((hash << 5) + hash) + c;
    }
    
    return hash;
}

// Usage:
unsigned int idx = hash_djb2(key) % size;
üêç Python
def hash_djb2(key: str) -> int:
    hash_value = 5381
    
    for char in key:
        hash_value = ((hash_value << 5) + 
                      hash_value) + ord(char)
    
    return hash_value

# Usage:

# Or use built-in:

Hash Table Structure

üìù Pseudocode
STRUCTURE Node:
    key: STRING
    value: ANY
    next: Node (or NULL)
END STRUCTURE

STRUCTURE HashTable:
    buckets: ARRAY of Node
    size: INTEGER
    count: INTEGER
END STRUCTURE

FUNCTION create_table(size):
    table ‚Üê NEW HashTable
    table.size ‚Üê size
    table.count ‚Üê 0
    table.buckets ‚Üê 
        NEW ARRAY[size] of NULL
    RETURN table
END FUNCTION
‚öôÔ∏è C
typedef struct Node {
    char *key;
    int value;
    struct Node *next;
} Node;

typedef struct {
    Node **buckets;
    int size;
    int count;
} HashTable;

HashTable *ht_create(int size) {
    HashTable *ht = malloc(
        sizeof(HashTable));
    ht->size = size;
    ht->count = 0;
    ht->buckets = calloc(size,
        sizeof(Node *));
    return ht;
}
üêç Python
class Node:
    def __init__(self, key, value):
        self.key = key
        self.value = value
        self.next = None

class HashTable:
    def __init__(self, size=16):
        self.size = size
        self.count = 0
        self.buckets = [None] * size
    
    @property
    def load_factor(self):
        return self.count / self.size

Insert Operation

üìù Pseudocode
FUNCTION insert(table, key, value):
    index ‚Üê hash(key) MOD table.size
    
    // Check for existing key
    current ‚Üê table.buckets[index]
    WHILE current ‚â† NULL:
        IF current.key = key:
            current.value ‚Üê value
            RETURN  // Updated
        END IF
        current ‚Üê current.next
    END WHILE
    
    // Insert new node at head
    new_node ‚Üê NEW Node
    new_node.key ‚Üê key
    new_node.value ‚Üê value
    new_node.next ‚Üê table.buckets[index]
    table.buckets[index] ‚Üê new_node
    table.count ‚Üê table.count + 1
END FUNCTION
‚öôÔ∏è C
void ht_insert(HashTable *ht,
               const char *key,
               int value) {
    unsigned int idx = 
        hash_djb2(key) % ht->size;
    
    // Check for existing key
    Node *curr = ht->buckets[idx];
    while (curr != NULL) {
        if (strcmp(curr->key, key) == 0) {
            curr->value = value;
            return;
        }
        curr = curr->next;
    }
    
    // Create and insert new node
    Node *node = malloc(sizeof(Node));
    node->key = strdup(key);
    node->value = value;
    node->next = ht->buckets[idx];
    ht->buckets[idx] = node;
    ht->count++;
}
üêç Python
def insert(self, key, value):
    index = hash(key) % self.size
    
    # Check for existing key
    current = self.buckets[index]
    while current is not None:
        if current.key == key:
            current.value = value
            return
        current = current.next

    node = Node(key, value)
    node.next = self.buckets[index]
    self.buckets[index] = node
    self.count += 1

Search Operation

üìù Pseudocode
FUNCTION search(table, key):
    index ‚Üê hash(key) MOD table.size
    
    current ‚Üê table.buckets[index]
    WHILE current ‚â† NULL:
        IF current.key = key:
            RETURN current.value
        END IF
        current ‚Üê current.next
    END WHILE
    
    RETURN NOT_FOUND
END FUNCTION

// Complexity:
// Average: O(1)
// Worst:   O(n) when all keys collide
‚öôÔ∏è C
int *ht_search(HashTable *ht,
               const char *key) {
    unsigned int idx = 
        hash_djb2(key) % ht->size;
    
    Node *curr = ht->buckets[idx];
    while (curr != NULL) {
        if (strcmp(curr->key, key) == 0) {
            return &curr->value;
        }
        curr = curr->next;
    }
    
    return NULL;  // Not found
}

// Usage:
int *val = ht_search(table, "alice");
if (val != NULL) {
    printf("Found: %d\n", *val);
}
üêç Python

Delete Operation

üìù Pseudocode
FUNCTION delete(table, key):
    index ‚Üê hash(key) MOD table.size
    
    current ‚Üê table.buckets[index]
    previous ‚Üê NULL
    
    WHILE current ‚â† NULL:
        IF current.key = key:
            IF previous = NULL:
                // First node in chain
                table.buckets[index] ‚Üê 
                    current.next
            ELSE:
                // Middle or end of chain
                previous.next ‚Üê current.next
            END IF
            FREE current
            table.count ‚Üê table.count - 1
            RETURN TRUE
        END IF
        previous ‚Üê current
        current ‚Üê current.next
    END WHILE
    
    RETURN FALSE  // Key not found
END FUNCTION
‚öôÔ∏è C
bool ht_delete(HashTable *ht,
               const char *key) {
    unsigned int idx = 
        hash_djb2(key) % ht->size;
    
    Node *curr = ht->buckets[idx];
    Node *prev = NULL;
    
    while (curr != NULL) {
        if (strcmp(curr->key, key) == 0) {
            if (prev == NULL) {
                ht->buckets[idx] = curr->next;
            } else {
                prev->next = curr->next;
            }
            free(curr->key);
            free(curr);
            ht->count--;
            return true;
        }
        prev = curr;
        curr = curr->next;
    }
    return false;
}
üêç Python
def delete(self, key):
    index = hash(key) % self.size
    current = self.buckets[index]
    prev = None
    
    while current is not None:
        if current.key == key:
            if prev is None:
                self.buckets[index] = \
                    current.next
            else:
                prev.next = current.next
            self.count -= 1
            return True
        prev = current
        current = current.next
    return False

Dynamic Resizing (Rehashing)

üìù Pseudocode
FUNCTION rehash(table):
    old_buckets ‚Üê table.buckets
    old_size ‚Üê table.size
    
    // Double the size
    table.size ‚Üê old_size √ó 2
    table.buckets ‚Üê 
        NEW ARRAY[table.size] of NULL
    table.count ‚Üê 0
    
    // Reinsert all entries
    FOR i FROM 0 TO old_size - 1:
        current ‚Üê old_buckets[i]
        WHILE current ‚â† NULL:
            insert(table, 
                   current.key, 
                   current.value)
            temp ‚Üê current
            current ‚Üê current.next
            FREE temp
        END WHILE
    END FOR
    
    FREE old_buckets
END FUNCTION
‚öôÔ∏è C
void ht_rehash(HashTable *ht) {
    Node **old = ht->buckets;
    int old_size = ht->size;
    
    // Create new larger table
    ht->size *= 2;
    ht->buckets = calloc(ht->size,
        sizeof(Node *));
    ht->count = 0;
    
    // Reinsert all entries
    for (int i = 0; i < old_size; i++) {
        Node *curr = old[i];
        while (curr != NULL) {
            ht_insert(ht, curr->key,
                      curr->value);
            Node *temp = curr;
            curr = curr->next;
            free(temp->key);
            free(temp);
        }
    }
    free(old);
}
üêç Python
def rehash(self):
    old_buckets = self.buckets
    
    # Double the size
    self.size *= 2
    self.buckets = [None] * self.size
    self.count = 0
    
    # Reinsert all entries
    for bucket in old_buckets:
        current = bucket
        while current is not None:
            self.insert(current.key,
                       current.value)
            current = current.next

Open Addressing (Double Hashing)

üìù Pseudocode
FUNCTION insert_open(table, key, val):
    index ‚Üê h1(key) MOD table.size
    step ‚Üê h2(key)
    
    WHILE table[index] ‚â† EMPTY AND
          table[index] ‚â† TOMBSTONE:
        IF table[index].key = key:
            table[index].value ‚Üê val
            RETURN
        END IF
        index ‚Üê (index + step) MOD size
    END WHILE
    
    table[index] ‚Üê (key, val)
    count ‚Üê count + 1
END FUNCTION

// h2(key) must be coprime to size
// to guarantee all slots are probed
‚öôÔ∏è C
void oht_insert(OpenHT *ht,
                const char *key,
                int value) {
    unsigned int idx = 
        h1(key) % ht->size;
    unsigned int step = h2(key);
    
    while (ht->entries[idx].key != NULL
        && !ht->entries[idx].deleted) {
        if (strcmp(ht->entries[idx].key,
                   key) == 0) {
            ht->entries[idx].value = value;
            return;
        }
        idx = (idx + step) % ht->size;
    }
    
    ht->entries[idx].key = strdup(key);
    ht->entries[idx].value = value;
    ht->entries[idx].deleted = false;
    ht->count++;
}

def insert_open(self, key, value):
    index = self.h1(key) % self.size
    step = self.h2(key)
    
    while (self.entries[index] is not None
           and not self.entries[index].deleted):
        if self.entries[index].key == key:
            self.entries[index].value = value
            return
        index = (index + step) % self.size
    
    self.entries[index] = Entry(key, value)
    self.count += 1

Practical Example: Word Counter

üìù Pseudocode
FUNCTION count_words(text):
    table ‚Üê create_table(1000)
    words ‚Üê split(text, whitespace)
    
    FOR EACH word IN words:
        word ‚Üê lowercase(word)
        word ‚Üê remove_punctuation(word)
        
        count ‚Üê search(table, word)
        IF count = NOT_FOUND:
            insert(table, word, 1)
        ELSE:
            insert(table, word, count + 1)
        END IF
    END FOR
    
    RETURN table
END FUNCTION

// Usage:
frequencies ‚Üê count_words(document)
top_10 ‚Üê get_highest(frequencies, 10)
‚öôÔ∏è C
HashTable *count_words(const char *text) {
    HashTable *ht = ht_create(1000);
    char *copy = strdup(text);
    char *word = strtok(copy, " \t\n");
    
    while (word != NULL) {
        // Convert to lowercase
        for (char *p = word; *p; p++) {
            *p = tolower(*p);
        }
        
        int *count = ht_search(ht, word);
        if (count == NULL) {
            ht_insert(ht, word, 1);
        } else {
            (*count)++;
        }
        word = strtok(NULL, " \t\n");
    }
    
    free(copy);
    return ht;
}
üêç Python
def count_words(text):
    counts = {}
    
    for word in text.split():
        word = word.lower().strip('.,!?')
        counts[word] = counts.get(word, 0) + 1
    
    return counts

# Even simpler with Counter:

    return Counter(words)
