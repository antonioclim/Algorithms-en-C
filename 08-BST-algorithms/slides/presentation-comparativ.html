üß± BST Node Structure

üìù Pseudocode
STRUCTURE BSTNode key: Integer left: Pointer to BSTNode right: Pointer to BSTNode END STRUCTURE // Abstract representation // No memory concerns // Focus on relationships
‚öôÔ∏è C
typedef struct BSTNode {
    int key;
    struct BSTNode *left;
    struct BSTNode *right;
} BSTNode;

/* Explicit pointer types */
/* Self-referential struct */
/* Size: 24 bytes (64-bit) */
üêç Python
class BSTNode:
    def __init__(self, key):
        self.key = key
        self.left = None
        self.right = None

# No pointer syntax
# References are implicit
# Memory managed by GC

üî® Creating a Node

üìù Pseudocode
FUNCTION CreateNode(key) node ‚Üê NEW BSTNode node.key ‚Üê key node.left ‚Üê NULL node.right ‚Üê NULL RETURN node END FUNCTION // Assumes allocation succeeds // No error handling shown
‚öôÔ∏è C
BSTNode *create_node(int key) {
    BSTNode *node = malloc(
        sizeof(BSTNode)
    );
    
    if (node == NULL) {
        return NULL;  /* Error! */
    }
    
    node->key = key;
    node->left = NULL;
    node->right = NULL;
    
    return node;
}
/* Must check malloc result */
/* Must free() later */

üîç Search Algorithm

üìù Pseudocode
FUNCTION Search(root, key) IF root = NULL THEN RETURN NULL END IF IF key = root.key THEN RETURN root ELSE IF key < root.key THEN RETURN Search(root.left, key) ELSE RETURN Search(root.right, key) END IF END FUNCTION
‚öôÔ∏è C
BSTNode *search(BSTNode *root, 
                int key) {
    if (root == NULL) {
        return NULL;
    }
    
    if (key == root->key) {
        return root;
    }
    
    if (key < root->key) {
        return search(root->left, 
                      key);
    }
    
    return search(root->right, key);
}
üêç Python
def search(root, key):
    if root is None:
        return None
    
    if key == root.key:
        return root
    elif key < root.key:
        return search(root.left, 
                      key)
    else:
        return search(root.right, 
                      key)

# 'is None' preferred over '== None'
# No pointer syntax needed

‚ûï Insert Algorithm

üìù Pseudocode
FUNCTION Insert(root, key) IF root = NULL THEN RETURN CreateNode(key) END IF IF key < root.key THEN root.left ‚Üê Insert(root.left, key) ELSE IF key > root.key THEN root.right ‚Üê Insert(root.right, key) END IF // Duplicates ignored RETURN root END FUNCTION
‚öôÔ∏è C
BSTNode *insert(BSTNode *root, 
                int key) {
    if (root == NULL) {
        return create_node(key);
    }
    
    if (key < root->key) {
        root->left = insert(
            root->left, key
        );
    } else if (key > root->key) {
        root->right = insert(
            root->right, key
        );
    }
    /* Duplicates ignored */
    
    return root;
}

üìã In-order Traversal

üìù Pseudocode
FUNCTION InOrder(root) IF root ‚â† NULL THEN InOrder(root.left) PRINT(root.key) InOrder(root.right) END IF END FUNCTION // L-N-R order // Produces sorted output // for valid BST
‚öôÔ∏è C
void inorder(BSTNode *root) {
    if (root == NULL) {
        return;
    }
    
    inorder(root->left);
    printf("%d ", root->key);
    inorder(root->right);
}

/* Side effect: prints */
/* No return value */
/* Stack grows with depth */
üêç Python
def inorder(root):
    if root is None:
        return
    
    inorder(root.left)
    print(root.key, end=' ')
    inorder(root.right)

# Or return a list:
def inorder_list(root):
    if root is None:
        return []
    return (inorder_list(root.left)
            + [root.key]
            + inorder_list(root.right))

‚ûñ Delete Algorithm

üìù Pseudocode
FUNCTION Delete(root, key) IF root = NULL THEN RETURN NULL END IF IF key < root.key THEN root.left ‚Üê Delete(root.left, key) ELSE IF key > root.key THEN root.right ‚Üê Delete(root.right, key) ELSE // Found node to delete IF root.left = NULL THEN RETURN root.right ELSE IF root.right = NULL THEN RETURN root.left END IF // Two children: get successor succ ‚Üê FindMin(root.right) root.key ‚Üê succ.key root.right ‚Üê Delete(root.right, succ.key) END IF RETURN root END FUNCTION
‚öôÔ∏è C
BSTNode *delete(BSTNode *root, int key) {
    if (root == NULL) return NULL;
    
    if (key < root->key) {
        root->left = delete(root->left, key);
    } else if (key > root->key) {
        root->right = delete(root->right, key);
    } else {
        /* Found: handle cases */
        if (root->left == NULL) {
            BSTNode *temp = root->right;
            free(root);  /* MUST free! */
            return temp;
        }
        if (root->right == NULL) {
            BSTNode *temp = root->left;
            free(root);
            return temp;
        }
        /* Two children */
        BSTNode *succ = find_min(root->right);
        root->key = succ->key;
        root->right = delete(root->right, 
                             succ->key);
    }
    return root;
}
üêç Python
def delete(root, key):
    if root is None:
        return None
    
    if key < root.key:
        root.left = delete(root.left, key)
    elif key > root.key:
        root.right = delete(root.right, key)
    else:
        # Found node to delete
        if root.left is None:
            return root.right
        if root.right is None:
            return root.left
        
        # Two children: get successor
        succ = find_min(root.right)
        root.key = succ.key
        root.right = delete(root.right, 
                            succ.key)
    
    return root

# No explicit free() needed!
# Garbage collector handles it

üßπ Memory Management

üìù Pseudocode
FUNCTION FreeTree(root) IF root ‚â† NULL THEN FreeTree(root.left) FreeTree(root.right) DEALLOCATE(root) END IF END FUNCTION // Post-order traversal // Children freed before parent // Assumes manual memory
‚öôÔ∏è C
void free_tree(BSTNode *root) {
    if (root == NULL) {
        return;
    }
    
    /* MUST be post-order! */
    free_tree(root->left);
    free_tree(root->right);
    free(root);  /* Free LAST */
}

/* Every malloc needs a free */
/* Memory leaks are serious */
/* Use Valgrind to check */

/* Wrong order causes: */
/* - Use after free */
/* - Memory corruption */
üêç Python
# No explicit free needed!

# Simply remove reference:
root = None

# Garbage collector handles it
# Reference counting + cycle detection

# Or use del for clarity:
del root

# Python handles memory
# automatically when objects
# become unreachable

# Focus on algorithm,
# not memory management
Key Insight: C requires explicit memory management with free(), whilst Python's garbage collector automates this process entirely.

üí° Key Insights

What Stays the Same

  • Algorithm logic is identical
  • Time complexity: O(log n) average
  • Space complexity: O(h) for recursion
  • BST property enforcement
  • Three deletion cases
  • Traversal orders (in/pre/post)

What Changes

  • Syntax and keywords
  • Memory management approach
  • Error handling patterns
  • Type declarations
  • Pointer/reference notation
  • Standard library support
The Big Picture: Once you understand the algorithm in pseudocode, implementing it in any language becomes a matter of translating syntax, not reinventing logic.
