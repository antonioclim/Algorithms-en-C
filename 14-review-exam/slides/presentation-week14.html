ğŸ’¾ Memory Optimisation Techniques

Space-Optimised LCS: O(min(m,n)) instead of O(mn) C 
int lcs_space_optimised(const char *X, const char *Y, int m, int n) {
    /* Ensure X is shorter for minimal space */
    if (m > n) {
        const char *t = X; X = Y; Y = t;
        int temp = m; m = n; n = temp;
    }
    
    int *prev = calloc(m + 1, sizeof(int));
    int *curr = calloc(m + 1, sizeof(int));
    
    for (int j = 1; j <= n; j++) {
        for (int i = 1; i <= m; i++) {
            if (X[i-1] == Y[j-1])
                curr[i] = prev[i-1] + 1;
            else
                curr[i] = (prev[i] > curr[i-1]) ? prev[i] : curr[i-1];
        }
        /* Swap pointers - O(1) operation */
        int *temp = prev; prev = curr; curr = temp;
    }
    
    int result = prev[m];
    free(prev); free(curr);
    return result;
}

#include <time.h>

typedef struct {
    const char *name;
    double total_time;

double benchmark_algorithm(void (*algo)(int*, int), int *data, int n, int runs) {
    struct timespec start, end;
    double total = 0.0;
    
    for (int r = 0; r < runs; r++) {
        int *copy = malloc(n * sizeof(int));
        memcpy(copy, data, n * sizeof(int));
        
        clock_gettime(CLOCK_MONOTONIC, &start);
        algo(copy, n);
        clock_gettime(CLOCK_MONOTONIC, &end);
        
        total += (end.tv_sec - start.tv_sec) + 
                 (end.tv_nsec - start.tv_nsec) / 1e9;
        free(copy);
    }
    
    return total / runs;
}

ğŸ›¤ï¸ Dijkstra's Algorithm

Priority Queue Implementation C 
void dijkstra(Graph *g, int src, int *dist, int *prev) {
    PriorityQueue *pq = pq_create(g->V);
    
    for (int v = 0; v < g->V; v++) {
        dist[v] = INT_MAX;
        prev[v] = -1;
    }
    dist[src] = 0;
    pq_insert(pq, src, 0);
    
    while (!pq_is_empty(pq)) {
        int u = pq_extract_min(pq);
        
        for (AdjNode *adj = g->lists[u]; adj; adj = adj->next) {
            int v = adj->vertex;
            int weight = adj->weight;
            
            if (dist[u] != INT_MAX && dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                prev[v] = u;
                pq_decrease_key(pq, v, dist[v]);
            }
        }
    }
    pq_destroy(pq);
}

ğŸŒ³ Minimum Spanning Tree Algorithms

Prim's Algorithm

  • Grows tree from single vertex
  • Uses priority queue
  • Time: O((V+E) log V)
  • Best for dense graphs

Strategy

Always add minimum-weight edge connecting tree to non-tree vertex.

Kruskal's Algorithm

  • Sorts all edges first
  • Uses Union-Find structure
  • Time: O(E log E)
  • Best for sparse graphs

Strategy

Add edges in weight order, skipping those that create cycles.

ğŸ”— Union-Find (Disjoint Set)

Path Compression and Union by Rank C 
typedef struct {
    int *parent;
    int *rank;
    int size;
} UnionFind;

/* Find with path compression - nearly O(1) amortised */
int uf_find(UnionFind *uf, int x) {
    if (uf->parent[x] != x)
        uf->parent[x] = uf_find(uf, uf->parent[x]);
    return uf->parent[x];
}

/* Union by rank - keeps trees shallow */
void uf_union(UnionFind *uf, int x, int y) {
    int rx = uf_find(uf, x);
    int ry = uf_find(uf, y);
    if (rx == ry) return;
    
    if (uf->rank[rx] < uf->rank[ry]) {
        uf->parent[rx] = ry;
    } else if (uf->rank[rx] > uf->rank[ry]) {
        uf->parent[ry] = rx;
    } else {
        uf->parent[ry] = rx;
        uf->rank[rx]++;
    }
}

/* djb2 hash function */
size_t djb2_hash(const char *str) {
    size_t hash = 5381;
    int c;
    while ((c = *str++))
        hash = ((hash << 5) + hash) + c;  /* hash * 33 + c */
    return hash;
}

void *ht_get(HashTable *ht, const char *key) {
    size_t idx = ht->hash(key) % ht->size;
    for (HashNode *n = ht->buckets[idx]; n; n = n->next)
        if (strcmp(n->key, key) == 0)
            return n->value;
    return NULL;
}

ğŸ“¦ Binary Heap Operations

Min-Heap with Heapify C 
void heapify_down(Heap *h, int i) {
    int smallest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    
    if (left < h->size && h->data[left] < h->data[smallest])
        smallest = left;
    if (right < h->size && h->data[right] < h->data[smallest])
        smallest = right;
    
    if (smallest != i) {
        swap(&h->data[i], &h->data[smallest]);
        heapify_down(h, smallest);
    }
}

void heapify_up(Heap *h, int i) {
    while (i > 0) {
        int parent = (i - 1) / 2;
        if (h->data[i] >= h->data[parent]) break;
        swap(&h->data[i], &h->data[parent]);
        i = parent;
    }
}

/* Build heap from array: O(n) */
void build_heap(Heap *h, int *arr, int n) {
    memcpy(h->data, arr, n * sizeof(int));
    h->size = n;
    for (int i = n/2 - 1; i >= 0; i--)
        heapify_down(h, i);
}

ğŸ’¼ Interview Preparation

"Describe BFS vs DFS" Common

BFS: O(branching^depth) space, level-by-level, shortest path in unweighted graphs

DFS: O(depth) space, follows branches, used for topological sort, cycle detection

"Detect cycle in linked list" Common

Floyd's tortoise and hare: O(n) time, O(1) space. Fast pointer moves 2x, slow moves 1x.

"Explain dynamic programming" Medium

Overlapping subproblems + optimal substructure. Memoisation (top-down) or tabulation (bottom-up).

"Implement priority queue" Medium

Binary heap: O(log n) insert/extract-min, O(1) find-min. Array-based with heapify operations.
