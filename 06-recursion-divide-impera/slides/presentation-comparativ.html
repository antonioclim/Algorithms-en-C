Queue Structure Definition

üìù Pseudocode

STRUCTURE Queue
    data: ARRAY[0..MAX-1] OF Element
    front: INTEGER
    rear: INTEGER
    count: INTEGER
    capacity: INTEGER
END STRUCTURE

// No memory management needed
// Focus on logical structure

‚öôÔ∏è C

typedef struct {
    int *data;      /* Dynamic array */
    int front;
    int rear;
    int count;
    int capacity;
} Queue;

Queue *queue_create(int cap) {
    Queue *q = malloc(sizeof(Queue));
    q->data = malloc(cap * sizeof(int));
    q->front = q->rear = q->count = 0;
    q->capacity = cap;
    return q;
}

void queue_destroy(Queue *q) {
    free(q->data);
    free(q);
}

Enqueue Operation

üìù Pseudocode

PROCEDURE Enqueue(Q, item)
    IF Q.count = Q.capacity THEN
        RETURN OVERFLOW_ERROR
    END IF
    
    Q.data[Q.rear] ‚Üê item
    Q.rear ‚Üê (Q.rear + 1) MOD Q.capacity
    Q.count ‚Üê Q.count + 1
    
    RETURN SUCCESS
END PROCEDURE

// Clear algorithm logic
// Modulo for wraparound

‚öôÔ∏è C

int enqueue(Queue *q, int item) {
    if (q->count >= q->capacity) {
        fprintf(stderr, "Queue overflow\n");
        return 0;  /* Failure */
    }
    
    q->data[q->rear] = item;
    q->rear = (q->rear + 1) % q->capacity;
    q->count++;
    
    return 1;  /* Success */
}

/* Caller must check return value */
if (!enqueue(&q, 42)) {
    /* Handle error */
}

üêç Python

def enqueue(self, item):
    if self.count >= self.capacity:
        raise OverflowError("Queue is full")
    
    self.data[self.rear] = item
    self.rear = (self.rear + 1) % self.capacity
    self.count += 1

# Or with deque (automatic):

q.append(43)  # Auto-discards oldest if full

# Exception handling:

    q.enqueue(item)

    print(f"Error: {e}")

Dequeue Operation

üìù Pseudocode

FUNCTION Dequeue(Q) ‚Üí Element
    IF Q.count = 0 THEN
        RETURN UNDERFLOW_ERROR
    END IF
    
    item ‚Üê Q.data[Q.front]
    Q.front ‚Üê (Q.front + 1) MOD Q.capacity
    Q.count ‚Üê Q.count - 1
    
    RETURN item
END FUNCTION

// Returns the removed element
// Front advances circularly

‚öôÔ∏è C

int dequeue(Queue *q, int *item) {
    if (q->count == 0) {
        fprintf(stderr, "Queue underflow\n");
        return 0;  /* Failure */
    }
    
    *item = q->data[q->front];
    q->front = (q->front + 1) % q->capacity;
    q->count--;
    
    return 1;  /* Success */
}

/* Usage with error checking */
int value;
if (dequeue(&q, &value)) {
    printf("Got: %d\n", value);
} else {
    printf("Queue was empty\n");
}

üêç Python

def dequeue(self):
    if self.count == 0:
        raise IndexError("Queue is empty")
    
    item = self.data[self.front]
    self.front = (self.front + 1) % self.capacity
    self.count -= 1
    
    return item

# With deque:
from collections import deque
q = deque([1, 2, 3])

item = q.popleft()  # Dequeue from front

# Safe dequeue:
try:
    item = q.popleft()
except IndexError:
    item = None  # or default value

Breadth-First Search

üìù Pseudocode

PROCEDURE BFS(Graph, start)
    visited ‚Üê ARRAY[vertices] OF FALSE
    Q ‚Üê NEW Queue
    
    visited[start] ‚Üê TRUE
    Enqueue(Q, start)
    
    WHILE NOT IsEmpty(Q) DO
        current ‚Üê Dequeue(Q)
        PROCESS(current)
        
        FOR EACH neighbour OF current DO
            IF NOT visited[neighbour] THEN
                visited[neighbour] ‚Üê TRUE
                Enqueue(Q, neighbour)
            END IF
        END FOR
    END WHILE
END PROCEDURE

üêç Python

from collections import deque

def bfs(graph, start):
    visited = set()
    queue = deque([start])
    visited.add(start)
    result = []
    
    while queue:
        current = queue.popleft()
        result.append(current)
        
        for neighbour in graph[current]:
            if neighbour not in visited:
                visited.add(neighbour)
                queue.append(neighbour)
    
    return result

# Usage:
graph = {0: [1, 3], 1: [0, 2], 2: [1], 3: [0]}
print(bfs(graph, 0))  # [0, 1, 3, 2]

Error Handling Approaches

üìù Pseudocode

// Abstract error indication

FUNCTION SafeDequeue(Q) ‚Üí (Element, Status)
    IF IsEmpty(Q) THEN
        RETURN (NULL, ERROR)
    ELSE
        RETURN (Dequeue(Q), OK)
    END IF
END FUNCTION

// Error handling left abstract
// Focus on algorithm flow

PRECONDITION: Q.count > 0
POSTCONDITION: Q.count = Q.count' - 1

‚öôÔ∏è C

/* Return codes */
#define SUCCESS 1
#define ERROR_EMPTY 0
#define ERROR_FULL -1

int dequeue(Queue *q, int *out) {
    if (!q || !out) return ERROR_EMPTY;
    if (q->count == 0) return ERROR_EMPTY;
    
    *out = q->data[q->front];
    q->front = (q->front + 1) % q->capacity;
    q->count--;
    return SUCCESS;
}

/* Caller responsibility */
int val;
int status = dequeue(&q, &val);
if (status != SUCCESS) {
    /* Handle error */
    fprintf(stderr, "Dequeue failed\n");
}

/* Using void pointers */
typedef struct {
    void **data;
    size_t elem_size;
    int front, rear, count, capacity;
} GenericQueue;

int gq_enqueue(GenericQueue *q, void *item) {
    if (q->count >= q->capacity) return 0;
    
    q->data[q->rear] = malloc(q->elem_size);
    memcpy(q->data[q->rear], item, q->elem_size);
    q->rear = (q->rear + 1) % q->capacity;
    q->count++;
    return 1;
}

Thread-Safe Queue

üìù Pseudocode

STRUCTURE ThreadSafeQueue
    queue: Queue
    lock: Mutex
    not_empty: Condition
    not_full: Condition
END STRUCTURE

PROCEDURE BlockingEnqueue(TSQ, item)
    ACQUIRE(TSQ.lock)
    WHILE IsFull(TSQ.queue) DO
        WAIT(TSQ.not_full, TSQ.lock)
    END WHILE
    
    Enqueue(TSQ.queue, item)
    SIGNAL(TSQ.not_empty)
    RELEASE(TSQ.lock)
END PROCEDURE

‚öôÔ∏è C

#include 

typedef struct {
    Queue queue;
    pthread_mutex_t lock;
    pthread_cond_t not_empty;
    pthread_cond_t not_full;
} ThreadSafeQueue;

void ts_enqueue(ThreadSafeQueue *tsq, int item) {
    pthread_mutex_lock(&tsq->lock);
    
    while (is_full(&tsq->queue)) {
        pthread_cond_wait(&tsq->not_full, &tsq->lock);
    }
    
    enqueue(&tsq->queue, item);
    pthread_cond_signal(&tsq->not_empty);
    pthread_mutex_unlock(&tsq->lock);
}

üí° Key Insights

üîë Core Principles Remain Constant

  • FIFO ordering is the same in every language
  • Circular buffer mathematics: (index + 1) % capacity
  • O(1) enqueue and dequeue complexity is achievable everywhere
  • Full vs empty distinction requires careful design in all languages

üéì Learning Path

  • Implement in C to master memory and pointers
  • Use Python for rapid problem-solving and verification
  • Compare implementations to deepen understanding
