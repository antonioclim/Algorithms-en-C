üìÇ Opening a File

üìù Pseudocode

OPEN file "data.txt" FOR reading IF file cannot be opened THEN PRINT error message EXIT program END IF // Use the file... CLOSE file

‚öôÔ∏è C

if (fp == NULL) {
    perror("Error opening file");
    return 1;
}

// Use the file...

fclose(fp);

üêç Python

# Context manager (recommended)
with open("data.txt", "r") as fp:
    # Use the file...
    pass
# Automatically closed!

# Or traditional way:
fp = open("data.txt", "r")
# Use the file...
fp.close()

üìñ Reading Lines from a File

üìù Pseudocode

OPEN file "input.txt" FOR reading WHILE not end of file DO READ line from file PROCESS line END WHILE CLOSE file

‚öôÔ∏è C

if (!fp) return 1;

// fgets returns NULL at EOF

    // Process line
    // Note: line includes '\n'

üêç Python

with open("input.txt", "r") as fp:
    # Iterate directly over file
    for line in fp:
        # Process line
        # Note: line includes '\n'
        print(line, end="")

# Or read all at once:
with open("input.txt") as fp:
    lines = fp.readlines()
    # or: content = fp.read()

üîç Parsing Structured Data

üìù Pseudocode

// Input: "1001 Alice 9.5" FOR EACH line IN file DO SPLIT line into parts id ‚Üê parts[0] AS integer name ‚Üê parts[1] AS string grade ‚Üê parts[2] AS float CREATE student(id, name, grade) ADD student to list END FOR

‚öôÔ∏è C

// Input: "1001 Alice 9.5"
char line[256];
while (fgets(line, sizeof(line), fp)) {
    int id;
    char name[50];
    float grade;
    
    // sscanf returns count matched
    if (sscanf(line, "%d %49s %f",
               &id, name, &grade) == 3) {
        students[count].id = id;
        strcpy(students[count].name, name);
        students[count].grade = grade;
        count++;
    }
}

üêç Python

# Input: "1001 Alice 9.5"
students = []
with open("data.txt") as fp:
    for line in fp:
        parts = line.split()
        if len(parts) >= 3:
            student = {
                "id": int(parts[0]),
                "name": parts[1],
                "grade": float(parts[2])
            }
            students.append(student)

# Or with unpacking:
id, name, grade = line.split()
grade = float(grade)

üìä CSV File Processing

üìù Pseudocode

// Input: "Alice,25,London" FOR EACH line IN csv_file DO fields ‚Üê SPLIT line BY "," name ‚Üê fields[0] age ‚Üê fields[1] AS integer city ‚Üê fields[2] PROCESS record(name, age, city) END FOR

‚öôÔ∏è C

// Manual CSV parsing
char line[256];
while (fgets(line, sizeof(line), fp)) {
    line[strcspn(line, "\n")] = '\0';
    
    char *name = strtok(line, ",");
    char *age_str = strtok(NULL, ",");
    char *city = strtok(NULL, ",");
    
    if (name && age_str && city) {
        int age = atoi(age_str);
        process_record(name, age, city);
    }
}
// Warning: strtok modifies string!

üêç Python

import csv

# Using csv module (handles quotes!)
with open("data.csv") as fp:
    reader = csv.reader(fp)
    for row in reader:
        name, age, city = row
        process_record(name, int(age), city)

# With DictReader (uses header row):
with open("data.csv") as fp:
    reader = csv.DictReader(fp)
    for row in reader:
        print(row["name"], row["age"])

üõ°Ô∏è Error Handling

üìù Pseudocode

TRY OPEN file "data.txt" PROCESS file contents CATCH FileNotFound PRINT "File does not exist" CATCH PermissionDenied PRINT "Access denied" FINALLY CLOSE file IF open END TRY

‚öôÔ∏è C

#include <errno.h>

FILE *fp = fopen("data.txt", "r");
if (fp == NULL) {
    // errno contains error code
    switch (errno) {
        case ENOENT:
            fprintf(stderr, "File not found\n");
            break;
        case EACCES:
            fprintf(stderr, "Access denied\n");
            break;
        default:
            perror("Error");  // Generic
    }
    return 1;
}
// ... use file ...
fclose(fp);

üêç Python

try:
    with open("data.txt") as fp:
        process(fp.read())
except FileNotFoundError:
    print("File does not exist")
except PermissionError:
    print("Access denied")
except IOError as e:
    print(f"I/O error: {e}")
finally:
    # Cleanup (not needed with 'with')
    pass

üéì Complete Example: Grade Calculator

üìù Pseudocode

FUNCTION calculate_averages(): students ‚Üê empty list OPEN "grades.txt" FOR reading FOR EACH line IN file DO PARSE name, grades from line avg ‚Üê SUM(grades) / COUNT(grades) ADD (name, avg) to students END FOR CLOSE file SORT students BY avg descending RETURN students

‚öôÔ∏è C

typedef struct {
    char name[50];
    float avg;
} Student;

int compare(const void *a, const void *b) {
    float diff = ((Student*)b)->avg 
               - ((Student*)a)->avg;
    return (diff > 0) - (diff < 0);
}

int load_students(Student *arr, int max) {
    FILE *fp = fopen("grades.txt", "r");
    if (!fp) return -1;
    
    char line[256];
    int n = 0;
    while (fgets(line, sizeof(line), fp) 
           && n < max) {
        // Parse and calculate...
        n++;
    }
    fclose(fp);
    qsort(arr, n, sizeof(Student), compare);
    return n;
}

üêç Python

def calculate_averages():
    students = []
    
    with open("grades.txt") as fp:
        for line in fp:
            parts = line.split()
            name = parts[0]
            grades = [float(g) 
                      for g in parts[1:]]
            avg = sum(grades) / len(grades)
            students.append({
                "name": name,
                "avg": avg
            })
    
    # Sort by average descending
    students.sort(key=lambda s: s["avg"],
                  reverse=True)
    return students

üîÑ File Iteration Patterns

üìù Pseudocode

// Character by character WHILE not EOF DO ch ‚Üê READ character PROCESS ch END WHILE // Line by line FOR EACH line IN file DO PROCESS line END FOR // Read all at once content ‚Üê READ ALL from file PROCESS content

‚öôÔ∏è C

// Character by character
int ch;
while ((ch = fgetc(fp)) != EOF) {
    putchar(toupper(ch));
}

// Line by line
char line[256];
while (fgets(line, sizeof(line), fp)) {
    process_line(line);
}

// Read all (need to know size)
fseek(fp, 0, SEEK_END);
long size = ftell(fp);
rewind(fp);
char *buf = malloc(size + 1);
fread(buf, 1, size, fp);
buf[size] = '\0';

üêç Python

# Character by character
with open("file.txt") as fp:
    while (ch := fp.read(1)):
        print(ch.upper(), end="")

# Line by line
with open("file.txt") as fp:
    for line in fp:
        process_line(line)

# Read all at once
with open("file.txt") as fp:
    content = fp.read()
    # or: lines = fp.readlines()

üí° Key Insights

1. Abstraction Level

C exposes the mechanics (buffers, pointers, manual cleanup). Python hides complexity behind elegant interfaces. Both approaches have merit.

2. Safety vs Control

Python's context managers eliminate resource leaks. C requires discipline but offers fine-grained control.

3. Error Handling Philosophy

C: check return values constantly. Python: ask forgiveness (exceptions) rather than permission.

4. The Algorithm Remains the Same

Regardless of language, the core logic is identical. Only syntax and memory management differ.
