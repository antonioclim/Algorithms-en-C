üí° Key Insight

In both C and Python, functions can be treated as "values" - stored in variables, passed to other functions, and returned from functions. This enables powerful patterns like callbacks, decorators, and higher-order functions.

C Approach

Functions have addresses in memory. A function pointer stores this address. Must declare exact signature.

Python Approach

Functions are first-class objects. Can be assigned to variables directly. No special syntax needed.

üìù Basic Callback Pattern (Pseudocode)

ALGORITHM: Generic Operation Application PSEUDOCODE 
ALGORITHM ApplyOperation
    INPUT: 
        x, y     : numbers
        operation: a function that takes two numbers and returns a number
    
    OUTPUT: 
        result   : the result of applying operation to x and y

    BEGIN
        result ‚Üê CALL operation(x, y)
        RETURN result
    END

---

EXAMPLE USAGE:

    FUNCTION add(a, b)
        RETURN a + b
    END FUNCTION
    
    FUNCTION multiply(a, b)
        RETURN a * b
    END FUNCTION
    
    // Using the generic algorithm with different operations
    result1 ‚Üê ApplyOperation(5, 3, add)        // result1 = 8
    result2 ‚Üê ApplyOperation(5, 3, multiply)   // result2 = 15
    
    // The SAME algorithm, DIFFERENT behavior based on callback!
Note: The pseudocode focuses on the logic - we pass a function as parameter. Implementation details (how to pass the function) differ by language.

‚ö° Basic Callback in C

C Implementation C 
#include <stdio.h>

// Define concrete operations
int add(int a, int b) { return a + b; }
int multiply(int a, int b) { return a * b; }

// Generic function that accepts a CALLBACK
// Note: must declare exact signature of function pointer
int apply_operation(int x, int y, int (*operation)(int, int)) {
    return operation(x, y);  // Call through the pointer
}

int main() {
    int result1 = apply_operation(5, 3, add);       // Pass function name
    int result2 = apply_operation(5, 3, multiply);
    
    printf("5 + 3 = %d\n", result1);   // Output: 5 + 3 = 8
    printf("5 * 3 = %d\n", result2);   // Output: 5 * 3 = 15
    
    return 0;
}

‚ö†Ô∏è C-Specific Requirements

  • Must declare the function pointer type: int (*operation)(int, int)
  • Parentheses around *operation are mandatory
  • Types must match exactly (int, int ‚Üí int)

üìù Generic Sorting (Pseudocode)

ALGORITHM: Generic Sort with Comparator PSEUDOCODE 
ALGORITHM GenericSort
    INPUT:
        array[]   : collection of elements
        n         : number of elements
        compare   : function(a, b) ‚Üí negative if a<b, 0 if equal, positive if a>b
    
    OUTPUT:
        array sorted according to compare function

    BEGIN
        // Using bubble sort for clarity (real implementations use quicksort)
        FOR i FROM 0 TO n-2
            FOR j FROM 0 TO n-2-i
                IF compare(array[j], array[j+1]) > 0 THEN
                    SWAP array[j], array[j+1]
                END IF
            END FOR
        END FOR
    END

---

COMPARATOR EXAMPLES:

    FUNCTION compare_ascending(a, b)
        RETURN a - b                    // negative if a < b
    END

    FUNCTION compare_descending(a, b)
        RETURN b - a                    // reversed comparison
    END

    FUNCTION compare_by_absolute(a, b)
        RETURN |a| - |b|                // compare absolute values
    END

‚ö° Sorting in C with qsort()

C Implementation using stdlib qsort() C 
#include <stdio.h>
#include <stdlib.h>

// Comparator functions - must match void* signature for qsort
int compare_asc(const void* a, const void* b) {
    return (*(int*)a - *(int*)b);
}

int compare_desc(const void* a, const void* b) {
    return (*(int*)b - *(int*)a);
}

int compare_abs(const void* a, const void* b) {
    return abs(*(int*)a) - abs(*(int*)b);
}

void print_array(int arr[], int n) {
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");
}

int main() {
    int arr[] = {-5, 2, -8, 3, 1, -2};
    int n = 6;
    
    printf("Original: "); print_array(arr, n);
    
    qsort(arr, n, sizeof(int), compare_asc);
    printf("Ascending: "); print_array(arr, n);
    
    qsort(arr, n, sizeof(int), compare_desc);
    printf("Descending: "); print_array(arr, n);
    
    int arr2[] = {-5, 2, -8, 3, 1, -2};
    qsort(arr2, n, sizeof(int), compare_abs);
    printf("By |value|: "); print_array(arr2, n);
    
    return 0;
}

üêç Sorting in Python

Python Implementation using sorted() and list.sort() PYTHON 
from functools import cmp_to_key

arr = [-5, 2, -8, 3, 1, -2]

# Method 1: Using 'key' function (preferred in Python)
print("Original:", arr)
print("Ascending:", sorted(arr))                           # Default
print("Descending:", sorted(arr, reverse=True))            # reverse flag
print("By |value|:", sorted(arr, key=abs))                 # key function

# Method 2: Using comparator (like C) - requires conversion
def compare_special(a, b):
    """Custom comparator: sort evens before odds"""
    if a % 2 == 0 and b % 2 != 0:
        return -1  # a (even) comes first
    if a % 2 != 0 and b % 2 == 0:
        return 1   # b (even) comes first
    return a - b   # same parity: sort by value

print("Evens first:", sorted(arr, key=cmp_to_key(compare_special)))

# Method 3: Lambda for inline key functions
students = [("Alice", 85), ("Bob", 92), ("Carol", 78)]
print("By name:", sorted(students, key=lambda s: s[0]))
print("By grade:", sorted(students, key=lambda s: s[1], reverse=True))

üí° Python's Approach

Python prefers key functions (return a value to sort by) over comparators. The cmp_to_key adapter converts C-style comparators when needed.

üìù Dispatch Table (Pseudocode)

PATTERN: Dispatch Table for Calculator PSEUDOCODE 
STRUCTURE: Dispatch Table
    A mapping from keys (operators) to functions (operations)
    
    dispatch_table = {
        '+' ‚Üí add_function,
        '-' ‚Üí subtract_function,
        '*' ‚Üí multiply_function,
        '/' ‚Üí divide_function
    }

ALGORITHM: Calculator
    INPUT: a, operator, b
    
    BEGIN
        IF operator IN dispatch_table THEN
            operation ‚Üê dispatch_table[operator]
            result ‚Üê CALL operation(a, b)
            RETURN result
        ELSE
            PRINT "Unknown operator"
            RETURN error
        END IF
    END

BENEFITS:
    1. Replaces long switch/if-else chains
    2. Easy to add new operations
    3. Operations can be added/removed at runtime
    4. Clean, maintainable code

‚ö° Dispatch Table in C

C Implementation with Function Pointer Array C 
#include <stdio.h>

// Define operation functions
int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }
int divide(int a, int b) { return b ? a / b : 0; }

// Type definition for cleaner code
typedef int (*Operation)(int, int);

// Dispatch table: array indexed by ASCII character
Operation dispatch[256] = {NULL};  // Initialize all to NULL

void init_dispatch() {
    dispatch['+'] = add;
    dispatch['-'] = sub;
    dispatch['*'] = mul;
    dispatch['/'] = divide;
}

int calculate(int a, char op, int b) {
    if (dispatch[(unsigned char)op] != NULL) {
        return dispatch[(unsigned char)op](a, b);
    }
    printf("Unknown operator: %c\n", op);
    return 0;
}

int main() {
    init_dispatch();
    
    printf("10 + 3 = %d\n", calculate(10, '+', 3));
    printf("10 - 3 = %d\n", calculate(10, '-', 3));
    printf("10 * 3 = %d\n", calculate(10, '*', 3));
    printf("10 / 3 = %d\n", calculate(10, '/', 3));
    
    return 0;
}

üêç Dispatch Table in Python

Python Implementation with Dictionary PYTHON 
# Define operations
def add(a, b): return a + b
def sub(a, b): return a - b
def mul(a, b): return a * b
def div(a, b): return a / b if b != 0 else float('inf')

# Dispatch table: dictionary mapping operators to functions
dispatch = {
    '+': add,
    '-': sub,
    '*': mul,
    '/': div,
}

def calculate(a, op, b):
    if op in dispatch:
        return dispatch[op](a, b)
    raise ValueError(f"Unknown operator: {op}")

# Usage
print(f"10 + 3 = {calculate(10, '+', 3)}")
print(f"10 - 3 = {calculate(10, '-', 3)}")
print(f"10 * 3 = {calculate(10, '*', 3)}")
print(f"10 / 3 = {calculate(10, '/', 3)}")

# Bonus: Add new operation at runtime!
dispatch['**'] = lambda a, b: a ** b
print(f"2 ** 10 = {calculate(2, '**', 10)}")

üîÑ Higher-Order Functions: Map/Filter/Reduce

Concept PSEUDOCODE 
MAP(array, func):
  result = []
  FOR each x IN array
    result.append(func(x))
  RETURN result

FILTER(array, predicate):
  result = []
  FOR each x IN array
    IF predicate(x) = true
      result.append(x)
  RETURN result

REDUCE(array, func, init):
  acc = init
  FOR each x IN array
    acc = func(acc, x)
  RETURN acc
C (Manual) C 
// Must implement manually
void map_int(int* in, int* out, 
             int n, int(*f)(int)) {
    for (int i = 0; i < n; i++)
        out[i] = f(in[i]);
}

// Filter
int filter_int(int* in, int* out,
               int n, int(*pred)(int)){
    int j = 0;
    for (int i = 0; i < n; i++)
        if (pred(in[i]))
            out[j++] = in[i];
    return j;
}

üìù Binary Search (Pseudocode)

ALGORITHM: Generic Binary Search PSEUDOCODE 
ALGORITHM BinarySearch
    INPUT:
        array[]   : SORTED collection
        n         : number of elements  
        key       : element to find
        compare   : function(key, element) ‚Üí negative/0/positive
    
    OUTPUT:
        index of found element, or -1 if not found

    BEGIN
        left ‚Üê 0
        right ‚Üê n - 1
        
        WHILE left ‚â§ right
            mid ‚Üê left + (right - left) / 2
            
            cmp_result ‚Üê compare(key, array[mid])
            
            IF cmp_result = 0 THEN
                RETURN mid          // Found!
            ELSE IF cmp_result < 0 THEN
                right ‚Üê mid - 1     // Search left half
            ELSE
                left ‚Üê mid + 1      // Search right half
            END IF
        END WHILE
        
        RETURN -1  // Not found
    END

NOTE: Array MUST be sorted according to same compare function!

üîç Binary Search: C vs Python

C with bsearch() C 
#include <stdlib.h>
#include <stdio.h>

int cmp(const void* key,
        const void* elem) {
    return *(int*)key - *(int*)elem;
}

int main() {
    int arr[] = {1, 3, 5, 7, 9, 11};
    int n = 6;
    int key = 7;
    
    int* found = bsearch(
        &key,        // key to find
        arr,         // sorted array
        n,           // count
        sizeof(int), // element size
        cmp          // comparator
    );
    
    if (found) {
        printf("Found %d at index %ld\n",
               *found, found - arr);
    } else {
        printf("Not found\n");
    }
    return 0;
}

üîí Closures: C vs Python

C: No Native Closures C 
// C doesn't have closures!
// Must pass context explicitly

typedef struct {
    int multiplier;
} Context;

int multiply_by(void* ctx, int x) {
    Context* c = (Context*)ctx;
    return x * c->multiplier;
}

// Using context
Context ctx = {.multiplier = 5};

// GLib provides closure-like API:
// GClosure* g_cclosure_new(...)
Python: Native Closures PYTHON 
# Closures capture environment!
def make_multiplier(n):
    # 'n' is captured by inner function
    def multiply(x):
        return x * n
    return multiply

times_5 = make_multiplier(5)
times_10 = make_multiplier(10)

print(times_5(3))   # 15
print(times_10(3))  # 30

# Each closure has its own 'n'!
# The captured variable persists

# Lambda closures work too:
make_adder = lambda n: lambda x: x + n
add_100 = make_adder(100)
print(add_100(5))  # 105

üí° Why Closures Matter

Closures allow functions to "remember" the environment in which they were created. In C, you must manually pass context; in Python, it's automatic.
