üì¶ Stack Structure Definition

üìù Pseudocode

STRUCTURE Stack:
    data: Array of Elements
    top: Integer  // -1 if empty
    capacity: Integer
END STRUCTURE

FUNCTION CreateStack(size):
    s ‚Üê new Stack
    s.data ‚Üê new Array[size]
    s.top ‚Üê -1
    s.capacity ‚Üê size
    RETURN s
END FUNCTION

‚öôÔ∏è C

typedef struct {
    int *data;
    int top;
    int capacity;
} Stack;

Stack* stack_create(int cap) {
    Stack *s = malloc(sizeof(Stack));
    if (!s) return NULL;
    
    s->data = malloc(cap * sizeof(int));
    if (!s->data) {
        free(s);
        return NULL;
    }
    s->top = -1;
    s->capacity = cap;
    return s;
}

‚¨ÜÔ∏è Push Operation

üìù Pseudocode

FUNCTION Push(stack, value):
    IF stack.top = stack.capacity - 1:
        // Option 1: Error
        ERROR "Stack overflow"
        // Option 2: Resize
        ResizeStack(stack)
    END IF
    
    stack.top ‚Üê stack.top + 1
    stack.data[stack.top] ‚Üê value
END FUNCTION

‚öôÔ∏è C

int stack_push(Stack *s, int val) {
    // Check for overflow
    if (s->top == s->capacity - 1) {
        // Resize: double capacity
        int new_cap = s->capacity * 2;
        int *new_data = realloc(
            s->data, 
            new_cap * sizeof(int)
        );
        if (!new_data) return -1;
        s->data = new_data;
        s->capacity = new_cap;
    }
    
    s->data[++s->top] = val;
    return 0;  // Success
}

üêç Python

def push(self, value):
    # Check capacity if bounded
    if self.capacity is not None:
        if len(self.data) >= self.capacity:
            raise OverflowError("Stack full")
    
    self.data.append(value)

# Or with plain list:
stack = []
stack.append(42)  # That's it!

# Python handles memory
# automatically - no malloc,
# no realloc, no worries!

‚¨áÔ∏è Pop Operation

üìù Pseudocode

FUNCTION Pop(stack):
    IF IsEmpty(stack):
        ERROR "Stack underflow"
    END IF
    
    value ‚Üê stack.data[stack.top]
    stack.top ‚Üê stack.top - 1
    RETURN value
END FUNCTION

FUNCTION IsEmpty(stack):
    RETURN stack.top = -1
END FUNCTION

‚öôÔ∏è C

int stack_pop(Stack *s) {
    if (s->top == -1) {
        fprintf(stderr, 
            "Error: Stack underflow\n");
        exit(EXIT_FAILURE);
    }
    return s->data[s->top--];
}

int stack_is_empty(Stack *s) {
    return s->top == -1;
}

// Safe version:
int stack_pop_safe(Stack *s, int *out) {
    if (s->top == -1) return -1;
    *out = s->data[s->top--];
    return 0;
}

def pop(self):

üëÅÔ∏è Peek and Cleanup

üìù Pseudocode

FUNCTION Peek(stack):
    IF IsEmpty(stack):
        ERROR "Stack empty"
    END IF
    RETURN stack.data[stack.top]
END FUNCTION

FUNCTION Destroy(stack):
    // Free resources
    DELETE stack.data
    DELETE stack
END FUNCTION

‚öôÔ∏è C

int stack_peek(Stack *s) {
    if (s->top == -1) {
        fprintf(stderr, "Empty\n");
        exit(EXIT_FAILURE);
    }
    return s->data[s->top];
}

void stack_destroy(Stack *s) {
    if (s) {
        free(s->data);  // Free array
        free(s);        // Free struct
    }
}

// CRITICAL: Must call destroy
// to avoid memory leaks!

üîç Bracket Matching

üìù Pseudocode

FUNCTION CheckBrackets(expr):
    stack ‚Üê CreateStack()
    
    FOR EACH char IN expr:
        IF char IS opening bracket:
            Push(stack, char)
        ELSE IF char IS closing:
            IF IsEmpty(stack):
                RETURN FALSE
            END IF
            top ‚Üê Pop(stack)
            IF NOT Matches(top, char):
                RETURN FALSE
            END IF
        END IF
    END FOR
    
    RETURN IsEmpty(stack)
END FUNCTION

‚öôÔ∏è C

int check_brackets(const char *s) {
    Stack *stk = stack_create(100);
    
    for (int i = 0; s[i]; i++) {
        char c = s[i];
        if (c=='(' || c=='[' || c=='{') {
            stack_push(stk, c);
        }
        else if (c==')' || c==']' || c=='}') {
            if (stack_is_empty(stk)) {
                stack_destroy(stk);
                return 0;
            }
            char top = stack_pop(stk);
            if (!matches(top, c)) {
                stack_destroy(stk);
                return 0;
            }
        }
    }
    int valid = stack_is_empty(stk);
    stack_destroy(stk);
    return valid;
}

üêç Python

def check_brackets(expr):
    stack = []
    pairs = {')':'(', ']':'[', '}':'{'}
    
    for char in expr:
        if char in '([{':
            stack.append(char)
        elif char in ')]}':
            if not stack:
                return False
            if stack.pop() != pairs[char]:
                return False
    
    return len(stack) == 0

# Usage:
print(check_brackets("{[()]}"))  # True
print(check_brackets("{[(])}"))  # False

# Python: 12 lines
# C: 25+ lines (plus helpers)

üßÆ Postfix Expression Evaluation

üìù Pseudocode

FUNCTION EvalPostfix(tokens):
    stack ‚Üê CreateStack()
    
    FOR EACH token IN tokens:
        IF token IS number:
            Push(stack, token)
        ELSE IF token IS operator:
            b ‚Üê Pop(stack)
            a ‚Üê Pop(stack)
            result ‚Üê Apply(token, a, b)
            Push(stack, result)
        END IF
    END FOR
    
    RETURN Pop(stack)
END FUNCTION

‚öôÔ∏è C

int eval_postfix(char *expr) {
    Stack *s = stack_create(64);
    char *token = strtok(expr, " ");
    
    while (token) {
        if (isdigit(token[0])) {
            stack_push(s, atoi(token));
        } else {
            int b = stack_pop(s);
            int a = stack_pop(s);
            int r;
            switch (token[0]) {
                case '+': r = a + b; break;
                case '-': r = a - b; break;
                case '*': r = a * b; break;
                case '/': r = a / b; break;
            }
            stack_push(s, r);
        }
        token = strtok(NULL, " ");
    }
    int result = stack_pop(s);
    stack_destroy(s);
    return result;
}

üêç Python

def eval_postfix(expr):
    stack = []
    ops = {
        '+': lambda a,b: a+b,
        '-': lambda a,b: a-b,
        '*': lambda a,b: a*b,
        '/': lambda a,b: a//b,
    }
    
    for token in expr.split():
        if token.lstrip('-').isdigit():
            stack.append(int(token))
        else:
            b, a = stack.pop(), stack.pop()
            stack.append(ops[token](a, b))
    
    return stack.pop()

# Usage:
result = eval_postfix("3 4 2 * +")
print(result)  # 11

üß† Memory Management Comparison

Aspect C Python
Allocation Manual: malloc(), calloc() Automatic: object creation
Deallocation Manual: free() Automatic: garbage collection
Resizing Manual: realloc() Automatic: list grows as needed
Memory Leaks Common bug if forget free() Rare (GC handles it)
Performance Optimal (direct control) Overhead from GC
Safety Prone to buffer overflow, dangling pointers Memory-safe by design

üí° Trade-off

C offers control and performance; Python offers safety and simplicity. Choose based on requirements.

‚ö†Ô∏è Error Handling Styles

üìù Pseudocode

// Option 1: Return codes
FUNCTION SafePop(stack, result):
    IF IsEmpty(stack):
        RETURN ERROR_EMPTY
    END IF
    result ‚Üê stack.data[stack.top]
    stack.top ‚Üê stack.top - 1
    RETURN SUCCESS
END FUNCTION

// Option 2: Exceptions
FUNCTION Pop(stack):
    IF IsEmpty(stack):
        THROW StackUnderflowError
    END IF
    // ... rest of code
END FUNCTION

‚öôÔ∏è C

// Return codes (common in C)
typedef enum {
    STACK_OK = 0,
    STACK_EMPTY = -1,
    STACK_FULL = -2
} StackResult;

StackResult stack_pop(Stack *s, int *out) {
    if (s->top == -1) {
        return STACK_EMPTY;
    }
    *out = s->data[s->top--];
    return STACK_OK;
}

// Usage:
int value;
if (stack_pop(s, &value) == STACK_OK) {
    printf("Popped: %d\n", value);
} else {
    printf("Stack was empty\n");
}

üéØ When to Use Which Language

Choose C When:

  • Performance is critical (systems programming)
  • Memory usage must be minimised
  • Working close to hardware (embedded, OS)
  • Need precise control over memory layout
  • Interfacing with existing C libraries
  • Building foundational infrastructure

Choose Python When:

  • Development speed matters more than runtime
  • Prototyping and experimentation
  • Data analysis, scripting, automation
  • Teaching algorithm concepts
  • When memory management bugs are costly
  • Integration with scientific libraries

üîë Key Takeaway

Master the algorithm in pseudocode first. Then translate to the appropriate language for your context. The logic transfers; only syntax and memory handling differ.

Key Insights

  • Pseudocode captures pure algorithm logic
  • C requires explicit memory management
  • Python abstracts memory, focuses on logic
  • Stack operations are O(1) in all implementations
  • Error handling style differs: return codes vs exceptions
  • Choose language based on project requirements

Understanding multiple perspectives deepens algorithmic thinking.
