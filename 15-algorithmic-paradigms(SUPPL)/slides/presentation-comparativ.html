D&C MergeSort (Pseudocode)

ALGORITHM: MergeSort PSEUDOCODE 
ALGORITHM MergeSort(A, left, right)
    INPUT:
        A[]    : array of comparable elements
        left   : starting index
        right  : ending index
    
    OUTPUT:
        A[] sorted in ascending order
    
    BEGIN
        IF left < right THEN
            mid ‚Üê (left + right) / 2
            
            // DIVIDE: Split into two halves
            MergeSort(A, left, mid)       // Sort left half
            MergeSort(A, mid + 1, right)  // Sort right half
            
            // CONQUER: Merge sorted halves
            Merge(A, left, mid, right)
        END IF
    END

---

PROCEDURE Merge(A, left, mid, right)
    BEGIN
        n1 ‚Üê mid - left + 1
        n2 ‚Üê right - mid
        
        // Create temporary arrays
        CREATE L[n1], R[n2]
        
        // Copy data to temp arrays
        FOR i FROM 0 TO n1-1
            L[i] ‚Üê A[left + i]
        FOR j FROM 0 TO n2-1
            R[j] ‚Üê A[mid + 1 + j]
        
        // Merge back into A[]
        i ‚Üê 0, j ‚Üê 0, k ‚Üê left
        WHILE i < n1 AND j < n2
            IF L[i] ‚â§ R[j] THEN
                A[k] ‚Üê L[i]
                i ‚Üê i + 1
            ELSE
                A[k] ‚Üê R[j]
                j ‚Üê j + 1
            END IF
            k ‚Üê k + 1
        END WHILE
        
        // Copy remaining elements
        WHILE i < n1
            A[k] ‚Üê L[i]; i++; k++
        WHILE j < n2
            A[k] ‚Üê R[j]; j++; k++
    END

üìà Complexity Analysis

T(n) = 2T(n/2) + O(n) ‚Üí O(n log n) time, O(n) auxiliary space

D&C MergeSort in C

C Implementation C 
#include <stdio.h>
#include <stdlib.h>

void merge(int arr[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    
    // Allocate temporary arrays
    int *L = (int*)malloc(n1 * sizeof(int));
    int *R = (int*)malloc(n2 * sizeof(int));
    
    // Copy data to temp arrays
    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];
    
    // Merge back
    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else              arr[k++] = R[j++];
    }
    
    // Copy remaining elements
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
    
    free(L); free(R);  // CRITICAL: Free allocated memory!
}

void merge_sort(int arr[], int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;  // Avoid overflow
        merge_sort(arr, left, mid);
        merge_sort(arr, mid + 1, right);
        merge(arr, left, mid, right);
    }
}

‚ö†Ô∏è C-Specific Considerations

  • Must allocate and free temporary arrays manually
  • Use left + (right - left) / 2 to avoid integer overflow
  • Pass array size implicitly through indices

D&C MergeSort in Python

Python Implementation PYTHON 
def merge_sort(arr):
    """Pythonic MergeSort using slicing."""
    if len(arr) <= 1:
        return arr
    
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])    # Slicing creates new lists
    right = merge_sort(arr[mid:])
    
    return merge(left, right)

def merge(left, right):
    """Merge two sorted lists."""
    result = []
    i = j = 0
    
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    
    # Extend with remaining elements (slicing is elegant!)
    result.extend(left[i:])
    result.extend(right[i:])
    return result

# Alternative: In-place version (more like C)
def merge_sort_inplace(arr, left=0, right=None):
    if right is None:
        right = len(arr) - 1
    if left < right:
        mid = (left + right) // 2
        merge_sort_inplace(arr, left, mid)
        merge_sort_inplace(arr, mid + 1, right)
        merge_inplace(arr, left, mid, right)

üí° Python Advantages

  • Slicing arr[:mid] creates copies automatically
  • list.extend() handles remaining elements elegantly
  • No manual memory management needed
  • Default parameters enable cleaner API

D&C QuickSort Partition (Side-by-Side)

Pseudocode CONCEPT 
FUNCTION Partition(A, low, high)
    pivot ‚Üê A[high]  // Last element
    i ‚Üê low - 1      // Smaller element index
    
    FOR j FROM low TO high - 1
        IF A[j] ‚â§ pivot THEN
            i ‚Üê i + 1
            SWAP A[i], A[j]
        END IF
    END FOR
    
    SWAP A[i+1], A[high]
    RETURN i + 1  // Pivot's final position
END
C (Lomuto) C 
int partition(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = low - 1;
    
    for (int j = low; j < high; j++) {
        if (arr[j] <= pivot) {
            i++;
            // Swap arr[i] and arr[j]
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
    // Place pivot in correct position
    int temp = arr[i + 1];
    arr[i + 1] = arr[high];
    arr[high] = temp;
    
    return i + 1;
}
Python PYTHON 
def partition(arr, low, high):
    pivot = arr[high]
    i = low - 1
    
    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            # Pythonic swap!
            arr[i], arr[j] = arr[j], arr[i]
    
    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    return i + 1

# Bonus: Functional approach
def quick_sort_functional(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    mid = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quick_sort_functional(left) + mid + quick_sort_functional(right)
Key Difference: Python's tuple unpacking a, b = b, a enables elegant swapping without temporary variables. The functional version uses list comprehensions for a completely different approach.

GREEDY Activity Selection (Pseudocode)

ALGORITHM: Activity Selection (Greedy) PSEUDOCODE 
ALGORITHM ActivitySelection(activities)
    INPUT:
        activities[] : array of (start_time, end_time) pairs
    
    OUTPUT:
        Maximum set of non-overlapping activities
    
    BEGIN
        // KEY INSIGHT: Sort by finish time (greedy choice)
        SORT activities BY end_time (ascending)
        
        selected ‚Üê empty list
        ADD activities[0] TO selected  // First activity always selected
        
        last_finish ‚Üê activities[0].end_time
        
        FOR i FROM 1 TO n-1
            IF activities[i].start_time ‚â• last_finish THEN
                // Activity doesn't overlap with last selected
                ADD activities[i] TO selected
                last_finish ‚Üê activities[i].end_time
            END IF
        END FOR
        
        RETURN selected
    END

---

GREEDY CHOICE PROPERTY:
    Selecting the activity that finishes earliest leaves maximum
    time for remaining activities. This local choice leads to
    a globally optimal solution.

PROOF SKETCH:
    Let A be an optimal solution. If A doesn't include the
    earliest-finishing activity a‚ÇÅ, we can replace A's first
    activity with a‚ÇÅ (it finishes no later), yielding another
    optimal solution that includes a‚ÇÅ.

GREEDY Activity Selection (C vs Python)

C Implementation C 
typedef struct {
    int start, end;
} Activity;

// Comparator for qsort
int cmp_by_end(const void *a, const void *b) {
    return ((Activity*)a)->end - ((Activity*)b)->end;
}

int activity_selection(Activity acts[], int n, 
                       int selected[]) {
    // Sort by finish time
    qsort(acts, n, sizeof(Activity), cmp_by_end);
    
    int count = 0;
    selected[count++] = 0;  // First activity
    int last_end = acts[0].end;
    
    for (int i = 1; i < n; i++) {
        if (acts[i].start >= last_end) {
            selected[count++] = i;
            last_end = acts[i].end;
        }
    }
    return count;
}

Python Implementation PYTHON 
def activity_selection(activities):
    """
    Select maximum non-overlapping activities.
    Input: list of (start, end) tuples
    """
    # Sort by finish time (Pythonic!)
    sorted_acts = sorted(activities, key=lambda x: x[1])
    
    selected = [sorted_acts[0]]
    last_end = sorted_acts[0][1]
    
    for act in sorted_acts[1:]:
        if act[0] >= last_end:
            selected.append(act)
            last_end = act[1]
    
    return selected

# Usage
activities = [
    (1, 4), (3, 5), (0, 6), (5, 7),
    (3, 9), (5, 9), (6, 10), (8, 11)
]
result = activity_selection(activities)
# Returns: [(1, 4), (5, 7), (8, 11)]

# One-liner (advanced):
def activity_select_functional(acts):
    from functools import reduce
    return reduce(
        lambda sel, a: sel + [a] if a[0] >= sel[-1][1] else sel,
        sorted(acts, key=lambda x: x[1])[1:],
        [sorted(acts, key=lambda x: x[1])[0]]
    )

  • C: Requires struct definition, comparator function, manual index tracking
  • Python: Native tuple support, sorted() with key, list slicing
  • Both achieve O(n log n) due to sorting

Three Implementations Compared PSEUDOCODE 
// VERSION 1: Naive Recursive ‚Äî O(2‚Åø) time, O(n) stack space
FUNCTION Fib_Naive(n)
    IF n ‚â§ 1 THEN RETURN n
    RETURN Fib_Naive(n-1) + Fib_Naive(n-2)  // Exponential! Recalculates same values
END

---

// VERSION 2: Memoised (Top-Down DP) ‚Äî O(n) time, O(n) space
FUNCTION Fib_Memo(n, memo)
    IF n ‚â§ 1 THEN RETURN n
    IF memo[n] IS NOT EMPTY THEN
        RETURN memo[n]  // Already computed!
    END IF
    memo[n] ‚Üê Fib_Memo(n-1, memo) + Fib_Memo(n-2, memo)
    RETURN memo[n]
END

---

// VERSION 3: Tabulated (Bottom-Up DP) ‚Äî O(n) time, O(1) space
FUNCTION Fib_Tab(n)
    IF n ‚â§ 1 THEN RETURN n
    
    prev2 ‚Üê 0  // F(0)
    prev1 ‚Üê 1  // F(1)
    
    FOR i FROM 2 TO n
        current ‚Üê prev1 + prev2
        prev2 ‚Üê prev1
        prev1 ‚Üê current
    END FOR
    
    RETURN prev1
END

---

C Implementation (All Three Versions) C 
// Version 1: Naive ‚Äî DON'T USE for n > 40!
long long fib_naive(int n) {
    if (n <= 1) return n;
    return fib_naive(n - 1) + fib_naive(n - 2);
}

// Version 2: Memoised with static array
long long memo[100] = {0};  // Global cache
long long fib_memo(int n) {
    if (n <= 1) return n;
    if (memo[n] != 0) return memo[n];
    memo[n] = fib_memo(n - 1) + fib_memo(n - 2);
    return memo[n];
}

// Version 2b: Memoised with dynamic allocation
long long fib_memo_dynamic(int n, long long *cache) {
    if (n <= 1) return n;
    if (cache[n] != -1) return cache[n];
    cache[n] = fib_memo_dynamic(n - 1, cache) 
             + fib_memo_dynamic(n - 2, cache);
    return cache[n];
}

// Version 3: Tabulated ‚Äî BEST for production code
long long fib_tab(int n) {
    if (n <= 1) return n;
    
    long long prev2 = 0, prev1 = 1, current;
    for (int i = 2; i <= n; i++) {
        current = prev1 + prev2;
        prev2 = prev1;
        prev1 = current;
    }
    return prev1;
}

// Benchmark comparison
void benchmark(int n) {
    clock_t start = clock();
    long long result = fib_tab(n);
    double time = (double)(clock() - start) / CLOCKS_PER_SEC;
    printf("F(%d) = %lld in %.6f seconds\n", n, result, time);
}

def fib_tab(n):
    if n <= 1:
        return n
    prev2, prev1 = 0, 1
    for _ in range(2, n + 1):
        prev2, prev1 = prev1, prev1 + prev2  # Tuple unpacking!
    return prev1

# BONUS: Generator for Fibonacci sequence
def fib_generator():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

DP 0/1 Knapsack (Pseudocode)

ALGORITHM: 0/1 Knapsack Dynamic Programming PSEUDOCODE 
ALGORITHM Knapsack_01(weights[], values[], n, capacity)
    INPUT:
        weights[] : weight of each item
        values[]  : value of each item
        n         : number of items
        capacity  : maximum weight capacity
    
    OUTPUT:
        Maximum value achievable
    
    BEGIN
        // Create DP table: dp[i][w] = max value using items 0..i-1 with capacity w
        CREATE dp[n+1][capacity+1]
        
        // Base case: 0 items or 0 capacity ‚Üí 0 value
        FOR w FROM 0 TO capacity
            dp[0][w] ‚Üê 0
        FOR i FROM 0 TO n
            dp[i][0] ‚Üê 0
        
        // Fill table bottom-up
        FOR i FROM 1 TO n
            FOR w FROM 1 TO capacity
                // Option 1: Don't take item i
                dp[i][w] ‚Üê dp[i-1][w]
                
                // Option 2: Take item i (if it fits)
                IF weights[i-1] ‚â§ w THEN
                    take_value ‚Üê values[i-1] + dp[i-1][w - weights[i-1]]
                    dp[i][w] ‚Üê MAX(dp[i][w], take_value)
                END IF
            END FOR
        END FOR
        
        RETURN dp[n][capacity]
    END

---

RECURRENCE RELATION:
    dp[i][w] = max(
        dp[i-1][w],                              // Don't take item i
        values[i-1] + dp[i-1][w - weights[i-1]]  // Take item i
    )

COMPLEXITY: O(n √ó capacity) time, O(n √ó capacity) space
SPACE-OPTIMISED: O(capacity) using single array (process backwards)

DP 0/1 Knapsack (C vs Python)

C Implementation C 
int knapsack_01(int weights[], int values[], 
                int n, int capacity) {
    // Allocate 2D DP table
    int **dp = malloc((n + 1) * sizeof(int*));
    for (int i = 0; i <= n; i++) {
        dp[i] = calloc(capacity + 1, sizeof(int));
    }
    
    // Fill table
    for (int i = 1; i <= n; i++) {
        for (int w = 1; w <= capacity; w++) {
            dp[i][w] = dp[i-1][w];  // Don't take
            
            if (weights[i-1] <= w) {
                int take = values[i-1] + 
                           dp[i-1][w - weights[i-1]];
                if (take > dp[i][w]) {
                    dp[i][w] = take;
                }
            }
        }
    }
    
    int result = dp[n][capacity];
    
    // Free memory
    for (int i = 0; i <= n; i++) free(dp[i]);
    free(dp);
    
    return result;
}

// Space-optimised version O(capacity)
int knapsack_space_opt(int w[], int v[], 
                       int n, int cap) {
    int *dp = calloc(cap + 1, sizeof(int));
    
    for (int i = 0; i < n; i++) {
        // MUST go backwards to avoid using updated values!
        for (int j = cap; j >= w[i]; j--) {
            int take = v[i] + dp[j - w[i]];
            if (take > dp[j]) dp[j] = take;
        }
    }
    
    int result = dp[cap];
    free(dp);
    return result;
}
Python Implementation PYTHON 
def knapsack_01(weights, values, capacity):
    """Standard 2D DP approach."""
    n = len(weights)
    
    # Create DP table (list comprehension!)
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]
    
    for i in range(1, n + 1):
        for w in range(1, capacity + 1):
            dp[i][w] = dp[i-1][w]  # Don't take
            
            if weights[i-1] <= w:
                take = values[i-1] + dp[i-1][w - weights[i-1]]
                dp[i][w] = max(dp[i][w], take)
    
    return dp[n][capacity]

def knapsack_space_opt(weights, values, capacity):
    """Space-optimised O(capacity) version."""
    dp = [0] * (capacity + 1)
    
    for i, (w, v) in enumerate(zip(weights, values)):
        # Process backwards!
        for j in range(capacity, w - 1, -1):
            dp[j] = max(dp[j], v + dp[j - w])
    
    return dp[capacity]

# With item reconstruction
def knapsack_with_items(weights, values, capacity):
    n = len(weights)
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]
    
    # Fill table
    for i in range(1, n + 1):
        for w in range(capacity + 1):
            if weights[i-1] <= w:
                dp[i][w] = max(dp[i-1][w],
                    values[i-1] + dp[i-1][w-weights[i-1]])
            else:
                dp[i][w] = dp[i-1][w]
    
    # Backtrack to find items
    items = []
    w = capacity
    for i in range(n, 0, -1):
        if dp[i][w] != dp[i-1][w]:
            items.append(i - 1)
            w -= weights[i - 1]
    
    return dp[n][capacity], items[::-1]
Critical Insight: In the space-optimised version, we iterate backwards through capacities. This ensures we don't use values that were already updated in the current iteration (would allow taking same item twice).
